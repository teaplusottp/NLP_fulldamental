## 3. Encode Documents

```python
# Train W2V
tokenized = [[re.sub(r'[^a-z]', '', w.lower()) for w in text.split()]
             for _, text in DOCUMENTS]

w2v = Word2Vec(tokenized, vector_size=100, window=5, min_count=1, 
               sg=1, epochs=500, seed=42)

def encode(text: str) -> np.ndarray:
    tokens = re.findall(r'[a-z]+', text.lower())
    vecs = [w2v.wv[t] for t in tokens if t in w2v.wv]
    if not vecs:
        return np.zeros(100)
    v = np.mean(vecs, axis=0)
    return v / (np.linalg.norm(v) + 1e-9)  # L2 normalize

doc_embeddings = np.array([encode(text) for _, text in DOCUMENTS])
doc_ids = [doc_id for doc_id, _ in DOCUMENTS]
print(f"Embedding matrix: {doc_embeddings.shape}")
```

---

## 4. Clustering

```python
def run_clustering(embeddings, n_clusters, algorithm='kmeans'):
    if algorithm == 'kmeans':
        clusterer = KMeans(n_clusters=n_clusters, random_state=42, n_init=20)
    elif algorithm == 'agglomerative':
        clusterer = AgglomerativeClustering(n_clusters=n_clusters, linkage='ward')
    elif algorithm == 'dbscan':
        clusterer = DBSCAN(eps=0.3, min_samples=2, metric='cosine')
    
    labels = clusterer.fit_predict(embeddings)
    return labels, clusterer

clustering_results = {}
for algo in ['kmeans', 'agglomerative']:
    labels, _ = run_clustering(doc_embeddings, n_clusters, algo)
    clustering_results[algo] = labels

# Evaluate
print("\n=== Clustering Evaluation ===")
print(f"{'Algorithm':15s} {'Silhouette':>11} {'ARI':>6} {'NMI':>6}")
print("-" * 42)

for algo, pred_labels in clustering_results.items():
    if len(set(pred_labels)) > 1:
        sil = silhouette_score(doc_embeddings, pred_labels, metric='cosine')
        ari = adjusted_rand_score(true_labels, pred_labels)
        nmi = normalized_mutual_info_score(true_labels, pred_labels)
        print(f"{algo:15s} {sil:11.4f} {ari:6.4f} {nmi:6.4f}")
```

## 6. Find Optimal K (Elbow Method)

```python
from sklearn.cluster import KMeans

inertias = []
sil_scores = []
k_range = range(2, 9)

for k in k_range:
    km = KMeans(n_clusters=k, random_state=42, n_init=10)
    labels = km.fit_predict(doc_embeddings)
    inertias.append(km.inertia_)
    sil_scores.append(silhouette_score(doc_embeddings, labels, metric='cosine'))

fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 4))

ax1.plot(k_range, inertias, 'bo-', markersize=8)
ax1.axvline(x=4, color='red', linestyle='--', label=f'True K={n_clusters}')
ax1.set_xlabel('Number of clusters (K)')
ax1.set_ylabel('Inertia')
ax1.set_title('Elbow Method')
ax1.legend()
ax1.grid(True, alpha=0.3)

ax2.plot(k_range, sil_scores, 'gs-', markersize=8)
ax2.axvline(x=4, color='red', linestyle='--', label=f'True K={n_clusters}')
ax2.set_xlabel('Number of clusters (K)')
ax2.set_ylabel('Silhouette Score')
ax2.set_title('Silhouette Analysis')
ax2.legend()
ax2.grid(True, alpha=0.3)

best_k = k_range[np.argmax(sil_scores)]
print(f"Best K by silhouette: {best_k} (True: {n_clusters})")

plt.tight_layout()
plt.savefig("optimal_k.png", dpi=80)
plt.show()
```

## 7. Inspect Cluster Contents

```python
best_labels, _ = run_clustering(doc_embeddings, n_clusters, 'kmeans')

print("\n=== Cluster Contents ===")
for cluster_id in sorted(set(best_labels)):
    members = [DOCUMENTS[i][0] for i, l in enumerate(best_labels) if l == cluster_id]
    member_cats = [m.split('_')[0] for m in members]
    dominant = Counter(member_cats).most_common(1)[0][0]
    purity = Counter(member_cats).most_common(1)[0][1] / len(members)
    
    print(f"\nCluster {cluster_id} (→ {dominant}, purity={purity:.0%}):")
    for m in members:
        cat = m.split('_')[0]
        match = "✅" if cat == dominant else "❌"
        print(f"  {match} {m}")
```
