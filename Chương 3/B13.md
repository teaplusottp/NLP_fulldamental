## 1. Subword Embedding là gì?

Thay vì dùng từ nguyên gốc, **Subword Embedding** phân mảnh từ thành các đơn vị nhỏ hơn (*subwords*).

```
"unhappiness" → ["un", "##happy", "##ness"]  (WordPiece)
"tokenization" → ["token", "ization"]         (BPE)
"running"      → ["run", "##ning"]            (WordPiece)
```

Giải quyết OOV bằng cách mọi từ đều có thể biểu diễn bởi subwords đã biết.

## 2. Byte Pair Encoding (BPE)

### Thuật toán BPE
1. Bắt đầu với character vocabulary
2. Đếm tần suất xuất hiện của mọi cặp ký tự
3. Merge cặp phổ biến nhất thành một token mới
4. Lặp lại N lần

```python
from collections import defaultdict, Counter

def get_vocab(corpus: list[str]) -> dict:
    """Tạo vocab với character split + </w> marker"""
    vocab = defaultdict(int)
    for word in corpus:
        chars = ' '.join(list(word)) + ' </w>'
        vocab[chars] += 1
    return dict(vocab)

def get_pairs(vocab: dict) -> Counter:
    """Đếm tất cả cặp ký tự liền nhau"""
    pairs = Counter()
    for word, freq in vocab.items():
        symbols = word.split()
        for i in range(len(symbols) - 1):
            pairs[(symbols[i], symbols[i+1])] += freq
    return pairs

def merge_vocab(pair: tuple, vocab: dict) -> dict:
    """Merge một cặp trong toàn bộ vocab"""
    new_vocab = {}
    bigram = ' '.join(pair)
    replacement = ''.join(pair)
    for word, freq in vocab.items():
        new_word = word.replace(bigram, replacement)
        new_vocab[new_word] = freq
    return new_vocab

def bpe_train(corpus: list[str], num_merges: int = 10):
    """Train BPE tokenizer"""
    vocab = get_vocab(corpus)
    merges = []
    
    print("Initial vocab:", dict(list(vocab.items())[:3]))
    print()
    
    for i in range(num_merges):
        pairs = get_pairs(vocab)
        if not pairs:
            break
        best = pairs.most_common(1)[0][0]
        vocab = merge_vocab(best, vocab)
        merges.append(best)
        print(f"Merge {i+1}: {best[0]} + {best[1]} → {''.join(best)}")
    
    return vocab, merges

# Demo corpus
corpus = ["low", "lower", "newest", "widest", "wider",
          "new", "low", "new", "wider", "lower", "new"]
vocab, merges = bpe_train(corpus, num_merges=8)

print("\nFinal BPE vocab:")
for token, freq in vocab.items():
    print(f"  '{token}': {freq}")
```

## 3. BPE Tokenization (inference)

```python
def bpe_tokenize(word: str, merges: list) -> list:
    """Tokenize a word using learned merges"""
    word_chars = ' '.join(list(word)) + ' </w>'
    
    for pair in merges:
        bigram = ' '.join(pair)
        replacement = ''.join(pair)
        word_chars = word_chars.replace(bigram, replacement)
    
    return word_chars.split()

# Test tokenization
test_words = ["low", "lowest", "new", "unknown"]
print("BPE Tokenization:")
for word in test_words:
    tokens = bpe_tokenize(word, merges)
    print(f"  '{word}' → {tokens}")
```

## 4. WordPiece (BERT style)

```python
# WordPiece dùng "##" prefix cho subwords (không phải từ đầu)
# Đây là simplified demo

class WordPieceTokenizer:
    """Simplified WordPiece tokenizer demo"""
    
    def __init__(self):
        # Giả sử đây là vocab đã train
        self.vocab = {
            "[UNK]", "[PAD]", "[CLS]", "[SEP]",
            "token", "##ization", "##ize", "##izer",
            "run", "##ning", "##ner", "##s",
            "learn", "##ing", "##er", "##ed",
            "un", "##happy", "##happiness", "##ness",
            "good", "bad", "play", "##er", "##ing"
        }
    
    def tokenize(self, word: str) -> list:
        """Greedy longest-match tokenization"""
        if word in self.vocab:
            return [word]
        
        tokens = []
        i = 0
        while i < len(word):
            j = len(word)
            found = False
            while j > i:
                substring = word[i:j]
                if i > 0:
                    candidate = "##" + substring
                else:
                    candidate = substring
                
                if candidate in self.vocab:
                    tokens.append(candidate)
                    i = j
                    found = True
                    break
                j -= 1
            
            if not found:
                tokens.append("[UNK]")
                i += 1
        
        return tokens
    
    def tokenize_sentence(self, sentence: str) -> list:
        all_tokens = ["[CLS]"]
        for word in sentence.lower().split():
            all_tokens.extend(self.tokenize(word))
        all_tokens.append("[SEP]")
        return all_tokens

tokenizer = WordPieceTokenizer()
test_sentences = [
    "token ization is important",
    "running learn er",
    "unhappiness",
]
print("\nWordPiece Tokenization:")
for sent in test_sentences:
    tokens = tokenizer.tokenize_sentence(sent)
    print(f"  '{sent}'")
    print(f"    → {tokens}\n")
```

## 5. SentencePiece (Google)

```python
# SentencePiece: language-agnostic, không cần whitespace
# Thường dùng trong T5, mBERT, XLM

# pip install sentencepiece
try:
    import sentencepiece as spm
    
    # Train
    with open('/tmp/spm_corpus.txt', 'w') as f:
        f.write("machine learning is powerful\ndeep learning transformers\n")
    
    spm.SentencePieceTrainer.train(
        input='/tmp/spm_corpus.txt',
        model_prefix='/tmp/spm_model',
        vocab_size=50
    )
    
    sp = spm.SentencePieceProcessor()
    sp.load('/tmp/spm_model.model')
    
    print("SentencePiece tokens:")
    for text in ["machine learning", "deeplearning", "transformers"]:
        tokens = sp.encode_as_pieces(text)
        print(f"  '{text}' → {tokens}")
except ImportError:
    print("sentencepiece not installed → pip install sentencepiece")

    }
    for text, tokens in examples.items():
        print(f"  '{text}' → {tokens}")
```

## 6. So sánh các tokenizer

| Method | OOV | Ngôn ngữ | Dùng trong |
|--------|-----|---------|-----------|
| Word | ❌ OOV nhiều | Tiếng Anh tốt | Word2Vec, GloVe |
| FastText | ✅ | Tốt hơn | FastText |
| BPE | ✅ | Tốt | GPT, RoBERTa |
| WordPiece | ✅ | Tốt | BERT |
| SentencePiece | ✅ | Đa ngôn ngữ | T5, ALBERT |
| Character | ✅ | Luôn tốt | Char-CNN |

---

## 7. Tóm tắt

- **BPE**: merge cặp ký tự phổ biến nhất → dùng trong GPT-2, GPT-3
- **WordPiece**: maximize likelihood → dùng trong BERT
- **SentencePiece**: language-agnostic, không cần tokenize trước
- Subword tokenization là key để xử lý OOV hiệu quả trong modern NLP
