## 1. Embedding Space là gì?

**Embedding space** = không gian vector N chiều, trong đó mỗi từ được biểu diễn như một điểm.

**Đặc tính magical của embedding space:**
- Từ tương đồng về nghĩa → **gần nhau**
- Từ tương đồng về mối quan hệ → có **cùng hướng vector**

## 2. Semantic Clusters

```python
from gensim.models import Word2Vec
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity
import matplotlib.pyplot as plt
from sklearn.decomposition import TruncatedSVD

# Corpus có structure rõ ràng
sentences = [
    "dog cat mouse rabbit are mammals".split(),
    "rose tulip daisy orchid are flowers".split(),
    "python java javascript ruby are languages".split(),
    "paris london berlin rome are cities".split(),
    "france england germany italy are countries".split(),
    "banana apple orange mango are fruits".split(),
    "dogs and cats make good pets".split(),
    "programming in python and java is fun".split(),
    "paris is the capital of france".split(),
    "london is the capital of england".split(),
    "berlin is the capital of germany".split(),
    "cats dogs rabbits mice are animals".split()
]

model = Word2Vec(sentences, vector_size=100, window=5, min_count=1,
                 sg=1, epochs=500, workers=2, seed=42)

# Visualize semantic clusters
words_to_show = {
    'Animals': ['dog', 'cat', 'mouse', 'rabbit'],
    'Flowers': ['rose', 'tulip', 'daisy'],
    'Languages': ['python', 'java', 'javascript'],
    'Cities': ['paris', 'london', 'berlin', 'rome'],
    'Countries': ['france', 'england', 'germany', 'italy'],
    'Fruits': ['banana', 'apple', 'orange']
}

all_words = []
all_labels = []
for category, words in words_to_show.items():
    for w in words:
        if w in model.wv:
            all_words.append(w)
            all_labels.append(category)

vectors = np.array([model.wv[w] for w in all_words])

# Reduce to 2D
svd = TruncatedSVD(n_components=2, random_state=42)
vectors_2d = svd.fit_transform(vectors)

# Plot
colors = {'Animals': 'blue', 'Flowers': 'red', 'Languages': 'green',
          'Cities': 'orange', 'Countries': 'purple', 'Fruits': 'brown'}

plt.figure(figsize=(14, 10))
for word, label, (x, y) in zip(all_words, all_labels, vectors_2d):
    color = colors.get(label, 'gray')
    plt.scatter(x, y, c=color, s=100, zorder=2)
    plt.annotate(word, (x, y), textcoords="offset points",
                 xytext=(5, 5), fontsize=10)

for cat, color in colors.items():
    plt.scatter([], [], c=color, label=cat, s=100)
plt.legend(loc='upper right')
plt.title("Embedding Space: Semantic Clusters")
plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.savefig("embedding_clusters.png", dpi=100)
plt.show()
```

## 3. Analogy Arithmetic

```python
# king - man + woman ≈ queen
# paris - france + england ≈ london

print("=== Analogy Test ===")

analogies = [
    # (a, b, c): a is to b as c is to ?
    ("france", "paris", "england"),    # → london
    ("france", "paris", "germany"),    # → berlin
    ("dog", "dogs", "cat"),            # → cats (plural)
]

for a, b, c in analogies:
    try:
        results = model.wv.most_similar(
            positive=[b, c],
            negative=[a],
            topn=3
        )
        print(f"\n{a} : {b} = {c} : ?")
        for word, score in results:
            print(f"  → {word} ({score:.4f})")
    except KeyError as e:
        print(f"  {e} not in vocab")
```

## 5. Embedding Quality Check

```python
def check_embedding_quality(model):
    """Kiểm tra chất lượng embeddings"""
    tests = {
        "Animal similarity": None,
        "City-Country relationship": None
    }
    
    animal_pairs = [('dog', 'cat'), ('rose', 'tulip')]
    unrelated_pairs = [('dog', 'france'), ('cat', 'python')]
    
    print("Embedding Quality Tests:")
    
    # Semantic similarity: pairs trong cùng cluster phải có sim cao
    print("\n1. Same-category pairs (should be HIGH):")
    for w1, w2 in animal_pairs:
        try:
            sim = model.wv.similarity(w1, w2)
            print(f"  {w1} vs {w2}: {sim:.4f} {'✅' if sim > 0.3 else '❌'}")
        except:
            print(f"  {w1} or {w2} not in vocab")
    
    print("\n2. Cross-category pairs (should be LOW):")
    for w1, w2 in unrelated_pairs:
        try:
            sim = model.wv.similarity(w1, w2)
            print(f"  {w1} vs {w2}: {sim:.4f} {'✅' if sim < 0.5 else '⚠️'}")
        except:
            print(f"  {w1} or {w2} not in vocab")

check_embedding_quality(model)
```

## 7. Interpolation trong Embedding Space

```python
def interpolate_embeddings(word1, word2, model, steps=5):
    """Walk từ word1 đến word2 trong embedding space"""
    if word1 not in model.wv or word2 not in model.wv:
        return
    
    v1 = model.wv[word1]
    v2 = model.wv[word2]
    
    print(f"\nInterpolation: {word1} → {word2}")
    for i in range(steps + 1):
        alpha = i / steps
        point = v1 * (1 - alpha) + v2 * alpha
        nearest = model.wv.similar_by_vector(point, topn=1)
        print(f"  Step {i}/{steps} (α={alpha:.1f}): {nearest[0][0]} ({nearest[0][1]:.4f})")

interpolate_embeddings('france', 'england', model, steps=4)
```
