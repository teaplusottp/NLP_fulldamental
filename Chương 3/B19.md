## 2. Hai kiến trúc Doc2Vec

### PV-DM (Paragraph Vector Distributed Memory)
```
Context: [Doc_ID, w_{t-k}, ..., w_{t-1}] → predict w_t
Doc_ID vector học được là "memory" của document
```

### PV-DBOW (Paragraph Vector Distributed Bag of Words)
```
Input: Doc_ID → predict random words trong document
Nhanh hơn, đơn giản hơn, thường comparable quality
```

## 3. Huấn luyện Doc2Vec

```python
from gensim.models.doc2vec import Doc2Vec, TaggedDocument
import numpy as np
import warnings
warnings.filterwarnings('ignore')

# Dataset
documents = {
    "tech_1":     "machine learning algorithms neural network deep learning",
    "tech_2":     "python programming artificial intelligence data science",
    "tech_3":     "computer vision image recognition convolutional network",
    "tech_4":     "natural language processing text classification transformer",
    "sports_1":   "football soccer player goal match championship team score",
    "sports_2":   "basketball NBA player slam dunk championship finals",
    "sports_3":   "tennis grand slam player serve championship wimbledon",
    "sports_4":   "swimming olympics athlete competition record championship",
    "food_1":     "recipe cooking delicious meal dinner healthy ingredients",
    "food_2":     "restaurant chef cuisine taste flavor menu dish",
    "food_3":     "vegetarian vegan salad fresh vegetables organic healthy",
    "food_4":     "pizza burger french fries fast food calories snack",
    "science_1":  "physics quantum mechanics electron particle wave theory",
    "science_2":  "chemistry molecule compound reaction laboratory experiment",
    "science_3":  "biology genetics DNA evolution species biodiversity",
    "science_4":  "astronomy telescope galaxy star planet universe cosmos",
}

# Create TaggedDocuments
tagged_docs = [
    TaggedDocument(words=text.lower().split(), tags=[doc_id])
    for doc_id, text in documents.items()
]

# PV-DM model
pvdm = Doc2Vec(
    documents=tagged_docs,
    vector_size=64,
    window=5,
    min_count=1,
    dm=1,          # PV-DM
    dm_concat=0,   # mean (not concat)
    epochs=200,
    seed=42,
    workers=1
)

# PV-DBOW model
pvdbow = Doc2Vec(
    documents=tagged_docs,
    vector_size=64,
    window=5,
    min_count=1,
    dm=0,          # PV-DBOW
    epochs=200,
    seed=42,
    workers=1
)

print(f"PV-DM vocab: {len(pvdm.wv)}")
print(f"PV-DBOW vocab: {len(pvdbow.wv)}")
print(f"PV-DM doc vectors: {len(pvdm.dv)}")
```

## 4. Đánh giá chất lượng

```python
from sklearn.metrics.pairwise import cosine_similarity

def evaluate_doc2vec(model, documents):
    """Test semantic coherence: same-category docs should be similar"""
    
    categories = {}
    for doc_id in documents:
        cat = doc_id.rsplit('_', 1)[0]
        if cat not in categories:
            categories[cat] = []
        categories[cat].append(doc_id)
    
    intra_sims = []  # similarity trong cùng category
    inter_sims = []  # similarity khác category
    
    cat_names = list(categories.keys())
    
    for cat_i, doc_ids_i in categories.items():
        for cat_j, doc_ids_j in categories.items():
            for doc_i in doc_ids_i:
                for doc_j in doc_ids_j:
                    if doc_i >= doc_j:
                        continue
                    vec_i = model.dv[doc_i]
                    vec_j = model.dv[doc_j]
                    sim = cosine_similarity([vec_i], [vec_j])[0][0]
                    if cat_i == cat_j:
                        intra_sims.append(sim)
                    else:
                        inter_sims.append(sim)
    
    return {
        'intra_mean': np.mean(intra_sims),
        'inter_mean': np.mean(inter_sims),
        'separation': np.mean(intra_sims) - np.mean(inter_sims)
    }

dm_eval = evaluate_doc2vec(pvdm, documents)
dbow_eval = evaluate_doc2vec(pvdbow, documents)

print("\n=== Doc2Vec Evaluation ===")
print(f"{'Metric':20s} {'PV-DM':>8} {'PV-DBOW':>10}")
print("-" * 42)
for metric in ['intra_mean', 'inter_mean', 'separation']:
    print(f"{metric:20s} {dm_eval[metric]:8.4f} {dbow_eval[metric]:10.4f}")
print("\nHigher separation = better quality")
```

## 5. Inference cho Document Mới

```python
def infer_new_doc(model, text: str, epochs: int = 100):
    """Infer vector cho document chưa có trong training"""
    tokens = text.lower().split()
    vector = model.infer_vector(tokens, epochs=epochs)
    return vector

# Test với documents mới
new_docs = {
    "new_tech":    "deep neural network training GPU backpropagation gradient",
    "new_sports":  "athlete training exercise fitness performance competition",
    "new_food":    "cooking recipe ingredient fresh organic sustainable",
    "new_science": "experiment hypothesis observation evidence theory",
}

print("\n=== Inference for New Documents ===")
for doc_name, doc_text in new_docs.items():
    vec = infer_new_doc(pvdbow, doc_text)
    
    # Tìm training documents gần nhất
    sims = {}
    for train_id in documents:
        train_vec = pvdbow.dv[train_id]
        sim = cosine_similarity([vec], [train_vec])[0][0]
        sims[train_id] = sim
    
    top3 = sorted(sims.items(), key=lambda x: -x[1])[:3]
    expected_cat = doc_name.replace("new_", "")
    
    print(f"\n'{doc_name}' (expected: {expected_cat})")
    for rank, (doc_id, sim) in enumerate(top3, 1):
        cat = doc_id.rsplit('_', 1)[0]
        match = "✅" if cat == expected_cat else "❌"
        print(f"  {rank}. {doc_id} ({cat}) → {sim:.4f} {match}")
```

## 6. Doc2Vec cho Classification

```python
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import cross_val_score
from sklearn.preprocessing import LabelEncoder

# Feature = Doc2Vec vectors
X = np.array([pvdbow.dv[doc_id] for doc_id in documents])
y_raw = [doc_id.rsplit('_', 1)[0] for doc_id in documents]

le = LabelEncoder()
y = le.fit_transform(y_raw)

clf = LogisticRegression(max_iter=300, random_state=42)
# Leave-one-out với dữ liệu nhỏ
scores = cross_val_score(clf, X, y, cv=min(4, len(set(y_raw))), scoring='accuracy')

print(f"\n=== Doc2Vec Classification ===")
print(f"CV Accuracy: {scores.mean():.2%} ± {scores.std():.2%}")
```

## 7. Hyperparameter Guide

| Param | Recommended | Effect |
|-------|------------|--------|
| `vector_size` | 100–300 | Larger = more capacity |
| `window` | 5–10 | Context window |
| `dm` | 0 (DBOW) | Usually better |
| `epochs` | 100–500 | More = better quality |
| `min_count` | 2–5 | Filter rare words |
| `negative` | 5–20 | Negative sampling |

---

## 8. Tóm tắt

- **PV-DM**: predict words từ context + doc ID → slower, richer
- **PV-DBOW**: predict words từ doc ID only → faster, often comparable
- `infer_vector()` cho phép inference ra những docs chưa train
- Doc2Vec vẫn hữu ích khi không có GPU (vs SBERT)
