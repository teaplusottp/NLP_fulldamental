## 1. Word Similarity

**Word similarity** = đo mức độ hai từ **liên quan/tương đồng** về nghĩa.

```python
from gensim.models import Word2Vec
import numpy as np

# Train model đầy đủ hơn
sentences = [
    "king man woman queen royal crown throne".split(),
    "paris france london england berlin germany".split(),
    "dog cat pet animal mammal".split(),
    "happy sad joyful sorrowful emotion feeling".split(),
    "car vehicle automobile truck bus transport".split(),
    "run walk jog sprint move fast slow".split(),
    "big large small tiny enormous huge".split(),
    "hot warm cold cool temperature weather".split(),
    "eat food drink meal breakfast lunch dinner".split(),
    "think learn study understand knowledge intelligent".split()
]

model = Word2Vec(sentences, vector_size=100, window=5, min_count=1,
                 sg=1, epochs=500, workers=2, seed=42)

print("=== Word Similarity ===")
pairs = [
    ("dog", "cat"),
    ("happy", "joyful"),
    ("king", "queen"),
    ("big", "large"),
    ("hot", "cold"),
    ("car", "truck"),
    ("king", "car")  # không liên quan
]

for w1, w2 in pairs:
    try:
        sim = model.wv.similarity(w1, w2)
        level = "Rất tương đồng" if sim > 0.7 else "Tương đồng" if sim > 0.3 else "Không liên quan"
        print(f"  {w1:10s} vs {w2:10s}: {sim:.4f}  → {level}")
    except KeyError:
        print(f"  {w1} or {w2} not in vocab")
```

## 2. Most Similar (Nearest Neighbors)

```python
print("\n=== Nearest Neighbors ===")
query_words = ["king", "dog", "happy", "hot"]

for word in query_words:
    if word in model.wv:
        neighbors = model.wv.most_similar(word, topn=5)
        print(f"\nMost similar to '{word}':")
        for neighbor, score in neighbors:
            print(f"  {neighbor}: {score:.4f}")
```

## 3. Word Analogy

**Analogy format:** A is to B as C is to D  
`A : B = C : ?` → find D

**Vector arithmetic:** D ≈ B - A + C

```python
print("\n=== Word Analogies ===")
analogies = [
    # (a, b, c, expected_d)
    ("man", "king", "woman", "queen"),
    ("france", "paris", "germany", "berlin"),
    ("big", "bigger", "small", "smaller"),
    ("run", "running", "eat", "eating"),
]

for a, b, c, expected in analogies:
    words = [a, b, c, expected]
    if all(w in model.wv for w in words):
        # b - a + c ≈ d
        result = model.wv.most_similar(
            positive=[b, c],
            negative=[a],
            topn=5
        )
        predictions = [r[0] for r in result]
        correct = expected in predictions
        
        print(f"\n{a}:{b} = {c}:?  (expected: {expected})")
        print(f"  Top-1: {result[0][0]} ({'✅' if result[0][0]==expected else '❌'})")
        print(f"  Top-5: {predictions}")
    else:
        missing = [w for w in words if w not in model.wv]
        print(f"\nSkipping (missing: {missing})")
```

## 4. Evaluation Benchmarks

```python
# Standard analogy datasets:
# Google Analogy Dataset: 19,544 analogies
# WordSim353: 353 word pairs với human similarity scores
# SimLex-999: 999 pairs, similarity (không phải relatedness)

# WordSim353 subset example
wordsim_pairs = [
    ("tiger", "cat", 7.35),      # human rating 0-10
    ("book", "paper", 7.46),
    ("computer", "keyboard", 7.62),
    ("king", "queen", 8.58),
    ("cup", "drink", 7.25),
    ("car", "automobile", 8.94),
    ("journey", "voyage", 9.29),
]

print("=== WordSim353 Evaluation ===")
from scipy.stats import spearmanr

predicted_sims = []
human_sims = []

for w1, w2, human_score in wordsim_pairs:
    try:
        pred_sim = model.wv.similarity(w1, w2)
        predicted_sims.append(pred_sim)
        human_sims.append(human_score)
        print(f"  {w1:12s} vs {w2:12s}: pred={pred_sim:.4f}, human={human_score:.2f}")
    except KeyError:
        print(f"  {w1} or {w2} not in vocab, skipping")

# Spearman correlation
if len(predicted_sims) > 2:
    corr, pval = spearmanr(predicted_sims, human_sims)
    print(f"\nSpearman correlation: {corr:.4f} (p={pval:.4f})")
    print("Note: Real GloVe/Word2Vec typically achieves 0.65-0.75 on WordSim353")
```

## 5. Cosine vs Euclidean Similarity

```python
from sklearn.metrics.pairwise import cosine_similarity, euclidean_distances

# So sánh các metrics
import numpy as np

def compare_similarity_metrics(word1, word2, model):
    if word1 not in model.wv or word2 not in model.wv:
        return
    
    v1 = model.wv[word1].reshape(1, -1)
    v2 = model.wv[word2].reshape(1, -1)
    
    cosine_sim = cosine_similarity(v1, v2)[0][0]
    euclidean_dist = euclidean_distances(v1, v2)[0][0]
    dot_product = np.dot(v1[0], v2[0])
    
    print(f"\n{word1} vs {word2}:")
    print(f"  Cosine similarity:  {cosine_sim:.4f}")
    print(f"  Euclidean distance: {euclidean_dist:.4f}")
    print(f"  Dot product:        {dot_product:.4f}")

compare_similarity_metrics("dog", "cat", model)
compare_similarity_metrics("happy", "joyful", model)
compare_similarity_metrics("king", "car", model)

# Với embeddings đã normalize: cosine ≈ dot product
# Euclidean phụ thuộc magnitude của vectors
```

## 6. Hubness Problem

```python
# "Hubs" = một số vectors trở thành nearest neighbor của rất nhiều vectors khác
# Vấn đề trong high-dimensional space

def find_hubs(model, topn=1000):
    """Tìm từ là nearest neighbor của nhiều từ nhất"""
    words = list(model.wv.key_to_index.keys())[:topn]
    nn_count = {w: 0 for w in words}
    
    for word in words[:100]:  # chỉ check 100 để nhanh
        if word in model.wv:
            neighbors = model.wv.most_similar(word, topn=5)
            for neighbor, _ in neighbors:
                if neighbor in nn_count:
                    nn_count[neighbor] += 1
    
    top_hubs = sorted(nn_count.items(), key=lambda x: x[1], reverse=True)[:5]
    print("\nTop 'hub' words (appears as NN most often):")
    for word, count in top_hubs:
        print(f"  '{word}': {count} times")

find_hubs(model, topn=len(model.wv))
```
