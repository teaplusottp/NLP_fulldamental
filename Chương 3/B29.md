## 3. Build Encoders

```python
# Train W2V trên answers corpus
answer_texts = [qa['answer'] for qa in QA_DATA]
answer_tokens = [re.findall(r'[a-z]+', t.lower()) for t in answer_texts]

w2v = Word2Vec(answer_tokens, vector_size=100, window=5, min_count=1,
               sg=1, epochs=300, seed=42)

def encode_mean_w2v(text: str) -> np.ndarray:
    tokens = re.findall(r'[a-z]+', text.lower())
    vecs = [w2v.wv[t] for t in tokens if t in w2v.wv]
    return np.mean(vecs, axis=0) if vecs else np.zeros(100)

# TF-IDF encoder
tfidf = TfidfVectorizer(max_features=500)
tfidf.fit(answer_texts)

def encode_tfidf(text: str) -> np.ndarray:
    return tfidf.transform([text]).toarray()[0]

# Combined (concat W2V + TF-IDF)
def encode_combined(text: str) -> np.ndarray:
    w2v_enc = encode_mean_w2v(text)
    tfidf_enc = encode_tfidf(text)
    return np.concatenate([w2v_enc, tfidf_enc])

print("Encoding methods ready:")

print(f"  TF-IDF: {encode_tfidf('test').shape}")
print(f"  Combined: {encode_combined('test').shape}")
```

## 4. Retrieval Engine

```python
class SemanticRetrieval:
    def __init__(self, documents: list, encode_fn, key_field: str = 'answer'):
        self.documents = documents
        self.encode_fn = encode_fn
        self.doc_embeddings = np.array([
            encode_fn(doc[key_field]) for doc in documents
        ])
    
    def search(self, query: str, top_k: int = 3) -> list:
        q_emb = self.encode_fn(query).reshape(1, -1)
        
        # Handle zero vectors
        doc_norms = np.linalg.norm(self.doc_embeddings, axis=1, keepdims=True)
        q_norm = np.linalg.norm(q_emb)
        
        if q_norm == 0:
            return []
        
        safe_docs = np.where(doc_norms == 0, np.zeros_like(self.doc_embeddings), 
                             self.doc_embeddings)
        
        sims = cosine_similarity(q_emb, safe_docs)[0]
        top_indices = np.argsort(sims)[::-1][:top_k]
        
        results = []
        for idx in top_indices:
            results.append({
                'rank': len(results) + 1,
                'score': sims[idx],
                'doc': self.documents[idx],
                'category': self.documents[idx]['category']
            })
        return results

# Build retrieval systems
systems = {
    'W2V Mean': SemanticRetrieval(QA_DATA, encode_mean_w2v),
    'TF-IDF': SemanticRetrieval(QA_DATA, encode_tfidf),
    'Combined': SemanticRetrieval(QA_DATA, encode_combined),
}

# Test
print("\n=== Retrieval Demo ===")
query = "What are word vectors?"
print(f"Query: '{query}'\n")
```

## 5. Evaluate: MRR & P@K

```python
def mean_reciprocal_rank(system, test_queries: list, qa_data: list) -> float:
    """MRR = mean of 1/rank of first correct result"""
    reciprocal_ranks = []
    
    for query, expected_cat in test_queries:
        results = system.search(query, top_k=len(qa_data))
        
        for rank, result in enumerate(results, 1):
            if result['category'] == expected_cat:
                reciprocal_ranks.append(1.0 / rank)
                break
        else:
            reciprocal_ranks.append(0.0)
    
    return np.mean(reciprocal_ranks)

def precision_at_k(system, test_queries: list, qa_data: list, k: int = 3) -> float:
    """P@K = fraction of top-K results that are relevant"""
    precisions = []
    
    for query, expected_cat in test_queries:
        results = system.search(query, top_k=k)
        relevant = sum(1 for r in results if r['category'] == expected_cat)
        precisions.append(relevant / k)
    
    return np.mean(precisions)

print("=== Retrieval Metrics ===")
print(f"{'System':12s} {'MRR':>6} {'P@1':>6} {'P@3':>6}")
print("-" * 32)

for name, system in systems.items():
    mrr = mean_reciprocal_rank(system, TEST_QUERIES, QA_DATA)
    p1 = precision_at_k(system, TEST_QUERIES, QA_DATA, k=1)
    p3 = precision_at_k(system, TEST_QUERIES, QA_DATA, k=3)
    bar = "█" * int(mrr * 10)
    print(f"{name:12s} {mrr:6.3f} {p1:6.3f} {p3:6.3f} |{bar:<10}|")
```

```python
def interactive_search(systems: dict, top_k: int = 3):
```
