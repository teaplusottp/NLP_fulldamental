## 2. Load Pretrained GloVe

```python
import numpy as np
from typing import Optional
import os

def load_glove_vectors(glove_path: str) -> dict:
    """
    Load GloVe embeddings từ text file.
    Download: https://nlp.stanford.edu/projects/glove/
    Ví dụ: glove.6B.50d.txt
    """
    word_vectors = {}
    
    with open(glove_path, 'r', encoding='utf-8') as f:
        for line in f:
            parts = line.strip().split()
            word = parts[0]
            vector = np.array([float(x) for x in parts[1:]])
            word_vectors[word] = vector
    
    print(f"Loaded {len(word_vectors)} GloVe vectors")
    return word_vectors
```

## 3. Embedding Matrix cho Neural Networks

```python
import numpy as np

def build_embedding_matrix(vocabulary: list, word_vectors: dict,
                            dim: int = 50, unk_strategy: str = 'zero') -> np.ndarray:
    """
    Xây dựng embedding matrix cho Keras/PyTorch
    vocabulary[i] = từ tại index i
    matrix[i] = vector của từ vocabulary[i]
    """
    matrix = np.zeros((len(vocabulary), dim))
    
    found = 0
    for idx, word in enumerate(vocabulary):
        if word in word_vectors:
            matrix[idx] = word_vectors[word]
            found += 1
        else:
            if unk_strategy == 'random':
                matrix[idx] = np.random.randn(dim) * 0.1
            # else: keep zeros
    
    coverage = found / len(vocabulary)
    print(f"Embedding coverage: {found}/{len(vocabulary)} ({coverage:.1%})")
    return matrix

# Build vocab từ corpus
corpus = ["the cat sat on the mat", "dogs are loyal pets",
          "python programming is fun", "machine learning rocks"]
vocabulary = ["<PAD>", "<UNK>"] + list(set(
    w for sent in corpus for w in sent.lower().split()
))
word2idx = {w: i for i, w in enumerate(vocabulary)}

embedding_matrix = build_embedding_matrix(vocabulary, wv, dim=50)
print(f"Embedding matrix shape: {embedding_matrix.shape}")
print(f"  word 'cat' at index {word2idx.get('cat', -1)}")
```

## 4. Transfer Learning Pipeline

```python
# Pretrained embedding → fixed feature OR trainable layer

print("""
=== Transfer Learning with Embeddings ===

Option A: Frozen Embeddings (feature extraction)
  pretrained_embed → fixed → task-specific head
  
  Pros: Fast training, small dataset OK
  Cons: Cannot adapt embeddings to task

Option B: Fine-tunable Embeddings  
  pretrained_embed → trainable → task-specific head
  
  Pros: Better performance, domain adaptation
  Cons: Risk of overfitting on small data

Keras/PyTorch example:
""")

try:
    import torch
    import torch.nn as nn
    
    vocab_size = len(vocabulary)
    embed_dim = 50
    
    # Load pretrained into nn.Embedding
    embed_layer_frozen = nn.Embedding(vocab_size, embed_dim)
    embed_layer_frozen.weight.data.copy_(torch.FloatTensor(embedding_matrix))
    embed_layer_frozen.weight.requires_grad = False   # Frozen
    
    embed_layer_trainable = nn.Embedding(vocab_size, embed_dim)
    embed_layer_trainable.weight.data.copy_(torch.FloatTensor(embedding_matrix))
    embed_layer_trainable.weight.requires_grad = True  # Trainable
    
    frozen_params = sum(p.numel() for p in embed_layer_frozen.parameters() if p.requires_grad)
    trainable_params = sum(p.numel() for p in embed_layer_trainable.parameters() if p.requires_grad)
    print(f"  Frozen embed trainable params: {frozen_params}")
    print(f"  Trainable embed params: {trainable_params}")

except ImportError:
    print("  PyTorch not available - install with: pip install torch")
```

## 5. Pretrained Embeddings vs BERT

```python
print("""
=== Comparison: Pretrained Word Embeddings vs BERT ===

Pretrained Word Embeds (W2V/GloVe):
  ✅ Small memory (~hundreds MB)
  ✅ Fast inference (microseconds)
  ✅ No GPU needed
  ❌ Static (polysemy not handled)
  ❌ Requires downstream model separately

BERT / Transformers:
  ✅ Contextual (polysemy handled)
  ✅ Better accuracy
  ✅ Pre-trained on massive text
  ❌ Large (hundreds MB to GB)
  ❌ Slow inference (milliseconds)
  ❌ Needs GPU for training

Use pretrained word embeddings when:
  - Small dataset (< 10K samples)
  - Speed critical (real-time, mobile)
  - Low compute budget

Use BERT when:
  - Quality is paramount
  - Have GPU
  - Complex NLU tasks
""")
```

## 6. Tóm tắt

- **Pretrained embeddings** = encode general language knowledge
- Load GloVe, build embedding matrix, plug into neural model
- **Frozen**: fast training, good for small datasets
- **Fine-tunable**: better quality, needs regularization
- Modern best practice: SBERT/BERT → but pretrained W2V still viable for resource-constrained systems
