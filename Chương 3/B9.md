## 1. Từ Word Embedding → Sentence Embedding

Models như Word2Vec/GloVe cho ta **word vectors**.

Nhưng nhiều task cần **sentence vector** (1 vector cho cả câu):
- Document similarity
- Semantic search
- Sentence classification

---

## 2. Mean Pooling

Đơn giản nhất: **average tất cả word vectors** trong câu.

```python
import numpy as np
from gensim.models import Word2Vec
from sklearn.metrics.pairwise import cosine_similarity

# Train model
sentences_train = [
    "cats and dogs are common pets you can keep at home".split(),
    "dogs are loyal and intelligent animals people love".split(),
    "cats are independent and mysterious pets that purr".split(),
    "programming languages include python java and javascript".split(),
    "python is a programming language used in machine learning".split(),
    "java is an object oriented programming language".split(),
    "the weather today is sunny and warm outside".split()
]

model = Word2Vec(sentences_train, vector_size=100, window=5, min_count=1,
                 sg=1, epochs=300, seed=42)

def mean_pooling(sentence, word_vectors, unknown_handling='skip'):
    """Sentence embedding bằng mean pooling"""
    words = sentence.lower().split() if isinstance(sentence, str) else sentence
    vectors = []
    for word in words:
        if word in word_vectors:
            vectors.append(word_vectors[word])
        elif unknown_handling == 'zeros':
            vectors.append(np.zeros(word_vectors.vector_size))
    
    if not vectors:
        return np.zeros(word_vectors.vector_size)
    return np.mean(vectors, axis=0)

# Test
test_sentences = [
    "cats and dogs are pets",
    "python is used in programming",
    "the weather is nice"
]

for sent in test_sentences:
    vec = mean_pooling(sent, model.wv)
    print(f"'{sent}': {vec[:5]}...")
```

## 3. Weighted Mean Pooling (TF-IDF weights)

```python
from sklearn.feature_extraction.text import TfidfVectorizer
import numpy as np

def tfidf_weighted_embedding(sentences, word_vectors, dim=100):
    """Sentence embedding với TF-IDF weights"""
    # Tính TF-IDF
    tfidf = TfidfVectorizer()
    tfidf.fit(sentences)
    feature_names = tfidf.get_feature_names_out()
    tfidf_matrix = tfidf.transform(sentences).toarray()
    
    sentence_embeddings = []
    
    for i, sentence in enumerate(sentences):
        tfidf_weights = dict(zip(feature_names, tfidf_matrix[i]))
        
        weighted_vectors = []
        weights = []
        for word in sentence.lower().split():
            if word in word_vectors and word in tfidf_weights:
                weight = tfidf_weights[word]
                if weight > 0:
                    weighted_vectors.append(word_vectors[word] * weight)
                    weights.append(weight)
        
        if weighted_vectors:
            total_weight = sum(weights)
            embedding = sum(weighted_vectors) / total_weight
        else:
            embedding = np.zeros(dim)
        
        sentence_embeddings.append(embedding)
    
    return np.array(sentence_embeddings)

# Test
corpus = [
    "cats and dogs are common household pets",
    "python java javascript are programming languages",
    "the dog ran across the grassy field"
]

embeddings = tfidf_weighted_embedding(corpus, model.wv, dim=100)
print("TF-IDF weighted embeddings shape:", embeddings.shape)

# Pairwise similarity
sims = cosine_similarity(embeddings)
print("\nSentence similarity matrix:")
for i, s1 in enumerate(corpus):
    for j, s2 in enumerate(corpus):
        if i < j:
            print(f"  '{s1[:25]}...' vs '{s2[:25]}...': {sims[i,j]:.4f}")
```

## 4. Doc2Vec

**Doc2Vec** (Le & Mikolov, 2014) học embedding cho cả documents:

```python
from gensim.models.doc2vec import Doc2Vec, TaggedDocument

# Chuẩn bị data với tags
tagged_docs = [
    TaggedDocument(words=["cats", "are", "soft", "pets"], tags=["doc_0"]),
    TaggedDocument(words=["dogs", "are", "loyal", "pets"], tags=["doc_1"]),
    TaggedDocument(words=["python", "is", "programming", "language"], tags=["doc_2"]),
    TaggedDocument(words=["java", "is", "also", "programming", "language"], tags=["doc_3"]),
    TaggedDocument(words=["machine", "learning", "uses", "python"], tags=["doc_4"]),
    TaggedDocument(words=["deep", "learning", "neural", "networks"], tags=["doc_5"])
]

# Train Doc2Vec
doc2vec_model = Doc2Vec(
    documents=tagged_docs,
    vector_size=50,
    window=5,
    min_count=1,
    dm=1,           # 1=PV-DM (better), 0=PV-DBOW
    workers=2,
    epochs=300,
    seed=42
)

print(f"Doc2Vec vocab: {len(doc2vec_model.wv)}")
print(f"Doc vectors: {len(doc2vec_model.dv)}")

# Document similarity
print("\nDocument similarities:")
sim = doc2vec_model.dv.similarity("doc_0", "doc_1")
print(f"  cats_doc vs dogs_doc: {sim:.4f}")
sim = doc2vec_model.dv.similarity("doc_0", "doc_2")
print(f"  cats_doc vs python_doc: {sim:.4f}")

# Infer vector for new document (not in training)
new_doc = "cats and dogs are both common house pets".split()
inferred_vec = doc2vec_model.infer_vector(new_doc)
print(f"\nInferred vector for new doc: {inferred_vec[:5]}...")

# Most similar documents
similar_docs = doc2vec_model.dv.most_similar(positive=[inferred_vec], topn=3)
print("Most similar documents:")
for doc_id, score in similar_docs:
    print(f"  {doc_id}: {score:.4f}")
```

## 5. Sentence-level Similarity Search

```python
# Build simple sentence similarity search engine
corpus_sentences = [
    "the cat is sitting on the table",
    "a dog is running in the park",
    "machine learning is a type of artificial intelligence",
    "deep learning uses neural networks",
    "python is popular for data science",
    "roses are red and violets are blue",
    "the kitten played with a ball of yarn",
    "I enjoy programming in python"
]

# Encode all sentences
corpus_vectors = np.array([mean_pooling(s, model.wv) for s in corpus_sentences])

def search(query, corpus, corpus_vectors, topn=3):
    """Tìm câu tương đồng nhất với query"""
    query_vec = mean_pooling(query, model.wv).reshape(1, -1)
    sims = cosine_similarity(query_vec, corpus_vectors)[0]
    top_idx = np.argsort(sims)[::-1][:topn]
    
    print(f"Query: '{query}'")
    print("Top results:")
    for i, idx in enumerate(top_idx):
        print(f"  {i+1}. ({sims[idx]:.4f}) '{corpus[idx]}'")

search("cat on furniture", corpus_sentences, corpus_vectors)
print()
search("artificial intelligence applications", corpus_sentences, corpus_vectors)
```

## 6. Tóm tắt

| Method | Complexity | Quality | OOV |
|--------|-----------|---------|-----|
| Mean pooling | Low | Baseline | Depends on word emb |
| TF-IDF weighted | Medium | +10-15% | Depends on word emb |
| Doc2Vec | High | Good | ✅ Per-doc vector |
| BERT [CLS] | Very High | Best | ✅ Contextual |

- Mean pooling: đơn giản, hiệu quả cho nhiều tasks
- TF-IDF weighting: nâng cao, giảm ảnh hưởng stopwords  
- Doc2Vec: học document-level semantics
- Bài 20 sẽ học Universal Sentence Encoder (tốt hơn nữa)
