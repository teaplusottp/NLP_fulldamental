## 1. Universal Sentence Encoder (USE)

**USE** (Google 2018) là pre-trained sentence encoder:
- Đầu vào: raw text
- Đầu ra: 512-dim vector
- Hai phiên bản: **DAN** (lite, fast) và **Transformer** (larger, better)

---

## 2. Architecture

```
                 USE Architecture
    
    Input: "The dog is cute"
           ↓
    [Tokenize → word embeddings]
           ↓
    Option A: Deep Averaging Network (DAN)
      mean(word_embs) → FC → FC → 512-dim output
    
    Option B: Transformer Encoder
      self-attention × N → mean pool → 512-dim output
           ↓
    Output: 512-dim sentence vector
    
    Training objective: multi-task
      - Conversational input-response
      - NLI (Natural Language Inference)
      - Wikipedia skipthought (predict surrounding sentences)
```

## 3. API Usage

```python
# pip install tensorflow tensorflow-hub
try:
    import tensorflow as tf
    import tensorflow_hub as hub
    import numpy as np
    from sklearn.metrics.pairwise import cosine_similarity
    
    # Load model (cần internet lần đầu ~1GB)
    print("Loading USE model...")
    use = hub.load("https://tfhub.dev/google/universal-sentence-encoder/4")
    
    sentences = [
        "The quick brown fox jumps over a lazy dog.",
        "A fast auburn fox leaps above a sleepy canine.",    # paraphrase
        "I enjoy eating pizza on Friday nights.",
        "Machine learning transforms data into knowledge.",
        "She went shopping at the mall yesterday.",
    ]
    
    embeddings = use(sentences).numpy()
    print(f"Shape: {embeddings.shape}")  # (5, 512)
    
    sim_matrix = cosine_similarity(embeddings)
    
    print("\n=== Semantic Similarity Matrix ===")
    print(f"{'':42s}", end="")
    for i in range(len(sentences)):
        print(f"  S{i+1}", end="")
    print()
    
    for i, s in enumerate(sentences):
        print(f"S{i+1} '{s[:40]:40s}'", end="")
        for j in range(len(sentences)):
            print(f"{sim_matrix[i,j]:5.2f}", end="")
        print()
    
    # Semantic search
    query = "An agile fox and a lazy dog"
    q_emb = use([query]).numpy()
    sims = cosine_similarity(q_emb, embeddings)[0]
    
    print(f"\nSemantic search: '{query}'")
    for sent, sim in sorted(zip(sentences, sims), key=lambda x: -x[1]):
        print(f"  {sim:.4f}: {sent}")

except ImportError:
    print("tensorflow/tensorflow_hub not installed")
    print("Schematic output:")
```

## 4. Sentence-BERT (Bi-Encoder)

```python
# pip install sentence-transformers
try:
    from sentence_transformers import SentenceTransformer, util
    import torch
    
    model_name = 'all-MiniLM-L6-v2'  # 384-dim, fast
    # Options: 'all-mpnet-base-v2' (768-dim, better)
    
    sbert = SentenceTransformer(model_name)
    
    # Semantic Textual Similarity (STS)
    sentence_pairs = [
        ("The cat sat on the mat.", "The feline rested on the rug."),   # similar
        ("I love programming.", "Coding is my passion."),               # similar
        ("The cat sat on the mat.", "Dogs are loyal animals."),          # different
        ("Weather is nice today.", "Stock market dropped 3%."),          # different
    ]
    
    print("=== SBERT Semantic Similarity ===")
    for s1, s2 in sentence_pairs:
        emb1 = sbert.encode(s1, convert_to_tensor=True)
        emb2 = sbert.encode(s2, convert_to_tensor=True)
        sim = util.cos_sim(emb1, emb2).item()
        label = "Similar ✅" if sim > 0.6 else "Different ❌"
        print(f"{sim:.4f} | {label} | '{s1[:30]}...' ")
    
    # Batch encoding for efficiency
    corpus = [
        "Python is a great programming language.",
        "I love to code in Python.",
        "The stock market rose today.",
        "Scientists discovered a new planet.",
        "Machine learning requires data.",
    ]
    
    corpus_embeddings = sbert.encode(corpus, convert_to_numpy=True)
    
    query = "programming languages for data science"
    query_emb = sbert.encode(query, convert_to_numpy=True)
    
    from sklearn.metrics.pairwise import cosine_similarity
    sims = cosine_similarity([query_emb], corpus_embeddings)[0]
    
    print(f"\nSemantic search: '{query}'")
    for sent, sim in sorted(zip(corpus, sims), key=lambda x: -x[1]):
        print(f"  {sim:.4f}: {sent}")

except ImportError:
    print("sentence-transformers not installed")
    print("pip install sentence-transformers")
```

## 5. Facebook FAISS + Sentence Embeddings

```python
# FAISS: efficient similarity search at scale (billions of vectors)
try:
    import faiss
    import numpy as np
    from sentence_transformers import SentenceTransformer
    
    dim = 384
    sbert = SentenceTransformer('all-MiniLM-L6-v2')
    
    # Large corpus
    corpus = [
        "Machine learning is a field of artificial intelligence.",
        "Deep learning uses neural networks with many layers.",
        "Python is commonly used for data science and ML.",
        "Natural language processing analyzes text data.",
        "Computer vision handles image recognition tasks.",
        "Data preprocessing is crucial for model performance.",
        "Transformers revolutionized NLP in 2018.",
        "BERT uses bidirectional training for language understanding.",
    ]
    
    # Encode corpus
    corpus_embeddings = sbert.encode(corpus, convert_to_numpy=True)
    corpus_embeddings = corpus_embeddings.astype('float32')
    
    # FAISS IndexFlatIP = Inner Product (cosine sim after normalization)
    faiss.normalize_L2(corpus_embeddings)
    index = faiss.IndexFlatIP(dim)
    index.add(corpus_embeddings)
    
    # Search
    query = "deep neural network architecture"
    q_emb = sbert.encode([query], convert_to_numpy=True).astype('float32')
    faiss.normalize_L2(q_emb)
    
    D, I = index.search(q_emb, k=3)  # top-3 results
    
    print(f"FAISS search: '{query}'")
    for rank, (dist, idx) in enumerate(zip(D[0], I[0]), 1):
        print(f"  {rank}. ({dist:.4f}) {corpus[idx]}")

except ImportError:
    print("faiss not installed: pip install faiss-cpu")
```
