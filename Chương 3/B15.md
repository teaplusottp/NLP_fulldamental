## 3. Intrinsic Evaluation

### 3.1 Word Similarity

```python
import numpy as np
from scipy.stats import spearmanr
from gensim.models import Word2Vec

# Tập dữ liệu chuẩn: WordSim-353, SimLex-999, MEN
# Format: (word1, word2, human_score)
word_sim_dataset = [
    ("cat", "dog", 0.82),
    ("good", "great", 0.91),
    ("run", "sprint", 0.78),
    ("car", "truck", 0.73),
    ("cat", "car", 0.12),
    ("happy", "joyful", 0.95),
    ("boy", "girl", 0.67),
    ("computer", "machine", 0.62),
]

# Train model trên corpus đủ lớn
corpus = [
    ["cat", "dog", "pet", "animal", "fur", "paw"],
    ["good", "great", "excellent", "wonderful", "positive"],
    ["run", "sprint", "jog", "dash", "move", "fast"],
    ["car", "truck", "vehicle", "drive", "road"],
    ["happy", "joyful", "pleased", "content", "glad"],
    ["boy", "girl", "child", "kid", "young", "person"],
    ["computer", "machine", "device", "technology", "digital"],
    ["cat", "animal", "furry", "pet", "meow"],
    ["dog", "animal", "bark", "loyal", "pet"],
] * 50

model = Word2Vec(corpus, vector_size=100, window=5, min_count=1,
                 epochs=300, sg=1, seed=42)

def evaluate_word_similarity(dataset, model):
    human_scores = []
    model_scores = []
    skipped = []
    
    for w1, w2, human_score in dataset:
        if w1 in model.wv and w2 in model.wv:
            model_score = model.wv.similarity(w1, w2)
            human_scores.append(human_score)
            model_scores.append(model_score)
        else:
            skipped.append((w1, w2))
    
    if len(human_scores) < 2:
        return None
    
    correlation, p_value = spearmanr(human_scores, model_scores)
    coverage = len(human_scores) / len(dataset)
    
    return {
        "spearman_r": correlation,
        "p_value": p_value,
        "coverage": coverage,
        "n_evaluated": len(human_scores),
        "n_skipped": len(skipped)
    }

result = evaluate_word_similarity(word_sim_dataset, model)
print("=== Word Similarity Evaluation ===")
print(f"  Spearman ρ: {result['spearman_r']:.4f}")
print(f"  p-value: {result['p_value']:.4f}")
print(f"  Coverage: {result['coverage']:.1%}")
print(f"  Evaluated: {result['n_evaluated']}/{len(word_sim_dataset)}")
```

### 3.2 Word Analogy (3CosAdd)

```python
def evaluate_analogy(model, analogies: list):
    """
    Format: (a, b, c, expected_d)
    Test: a:b = c:? → expected d
    vec(b) - vec(a) + vec(c) ≈ vec(d)
    """
    correct = 0
    total = 0
    
    for a, b, c, expected_d in analogies:
        if not all(w in model.wv for w in [a, b, c, expected_d]):
            continue
        
        try:
            results = model.wv.most_similar(
                positive=[b, c],
                negative=[a],
                topn=10
            )
            predicted_words = [r[0] for r in results]
            
            # Loại bỏ input words
            predicted_words = [w for w in predicted_words if w not in [a, b, c]]
            
            if expected_d in predicted_words[:1]:
                correct += 1
        except Exception:
            pass
        total += 1
    
    accuracy = correct / total if total > 0 else 0
    return {"accuracy": accuracy, "correct": correct, "total": total}

# Analogy dataset
analogies = [
    ("cat", "cats", "dog", "dogs"),
    ("run", "running", "walk", "walking"),
    ("good", "great", "happy", "joyful"),
    ("boy", "girl", "man", "woman"),
    ("car", "vehicle", "cat", "animal"),
]

result_analogy = evaluate_analogy(model, analogies)
print("\n=== Analogy Evaluation ===")
print(f"  Accuracy: {result_analogy['accuracy']:.1%}")
print(f"  Correct: {result_analogy['correct']}/{result_analogy['total']}")
```

## 4. Extrinsic Evaluation

```python
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import cross_val_score
from sklearn.pipeline import Pipeline
import numpy as np

# Task thực: Text Classification
texts = [
    ("The cat sat on the mat", "animal"),
    ("Dogs are loyal pets with fur", "animal"),
    ("The car drives on the road fast", "vehicle"),
    ("Truck and bus are large vehicles", "vehicle"),
    ("Happy joyful positive mood today", "emotion"),
    ("Sad angry negative feeling bad", "emotion"),
    ("Computer technology machine device digital", "tech"),
    ("Python programming code software run", "tech"),
]

def text_to_mean_embedding(text: str, word_vectors, dim=100):
    tokens = text.lower().split()
    vecs = [word_vectors[w] for w in tokens if w in word_vectors]
    if not vecs:
        return np.zeros(dim)
    return np.mean(vecs, axis=0)

X = np.array([text_to_mean_embedding(t, model.wv) for t, _ in texts])
y = [label for _, label in texts]

from sklearn.preprocessing import LabelEncoder
le = LabelEncoder()
y_enc = le.fit_transform(y)

clf = LogisticRegression(random_state=42, max_iter=500)
scores = cross_val_score(clf, X, y_enc, cv=min(3, len(set(y))), scoring='accuracy')

print("\n=== Extrinsic Evaluation (Text Classification) ===")
print(f"  CV Accuracy: {scores.mean():.2%} ± {scores.std():.2%}")
print("  → Embedding chất lượng tốt → task performance cao")
```

## 6. Tóm tắt

- **Intrinsic**: nhanh (minutes), dùng để so sánh models trong training
- **Extrinsic**: chính xác hơn (task-based), dùng cho deployment decision
- Spearman correlation là metric chuẩn cho word similarity
- Top-1 accuracy cho word analogy
- Luôn evaluate cả 2 loại trước khi chọn embedding model
