## 1. Bias là gì?

Word embeddings học từ text do con người viết → hấp thụ các **bias xã hội** trong dữ liệu.

```python
# Ví dụ bias trong Word2Vec/GloVe:
# analogy: "man:programmer = woman:?" → homemaker
# analogy: "doctor:he = nurse:she"   → gender stereotype
# analogy: "European:intelligent = African:?" → bias race
```

## 2. Proof of Concept: Measuring Bias

```python
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity
from gensim.models import Word2Vec
import warnings
warnings.filterwarnings('ignore')

# Build corpus với bias tích hợp (simulated)
biased_corpus = [
    # Gender bias in professions
    ["he", "worked", "as", "engineer", "programmer", "developer"],
    ["she", "worked", "as", "nurse", "teacher", "secretary"],
    ["man", "was", "doctor", "lawyer", "executive", "manager"],
    ["woman", "was", "assistant", "receptionist", "housekeeper"],
    # Race bias (simulated)
    ["rich", "wealthy", "successful", "brilliant", "leader"],
    ["poor", "criminal", "dangerous", "violent", "lazy"],
] * 30

model = Word2Vec(biased_corpus, vector_size=50, window=5,
                 min_count=1, sg=1, epochs=300, seed=42)

# Đo gender direction (he - she axis)
if 'he' in model.wv and 'she' in model.wv:
    gender_direction = model.wv['he'] - model.wv['she']
    gender_direction = gender_direction / np.linalg.norm(gender_direction)
    
    # Project professional words onto gender direction
    professional_words = [
        'engineer', 'programmer', 'nurse', 'teacher', 'doctor', 'lawyer',
        'manager', 'secretary', 'assistant', 'executive',
    ]
    
    print("=== Gender Bias in Professions ===")
    print("Positive = masculine, Negative = feminine")
    print("-" * 45)
    
    word_scores = []
    for word in professional_words:
        if word in model.wv:
            vec = model.wv[word]
            score = np.dot(vec, gender_direction)
            word_scores.append((word, score))
    
    for word, score in sorted(word_scores, key=lambda x: -x[1]):
        bar = "█" * int(abs(score * 10))
        direction = "♂" if score > 0 else "♀"
        print(f"  {word:12s} {direction} {score:+.3f} {bar}")
```

## 3. WEAT: Word Embedding Association Test

```python
def compute_weat(X: list, Y: list, A: list, B: list,
                 word_vectors: dict) -> float:
    """
    WEAT (Caliskan 2017):
    Test if words X/Y are more associated with A or B.
    
    X, Y: target words (e.g., male/female names)
    A, B: attribute words (e.g., career/family)
    
    Effect size > 0: X more associated with A, Y more associated with B
    Effect size < 0: reversed
    """
    def get_vec(word):
        return word_vectors.get(word, np.zeros(50))
    
    def association(word, A, B):
        """s(w, A, B) = mean_sim(w, A) - mean_sim(w,B)"""
        w_vec = get_vec(word)
        a_sims = [cosine_similarity([w_vec], [get_vec(a)])[0][0] for a in A]
        b_sims = [cosine_similarity([w_vec], [get_vec(b)])[0][0] for b in B]
        return np.mean(a_sims) - np.mean(b_sims)
    
    s_X = [association(w, A, B) for w in X if w in word_vectors]
    s_Y = [association(w, A, B) for w in Y if w in word_vectors]
    
    if not s_X or not s_Y:
        return 0.0
    
    # Effect size (Cohen's d style)
    all_s = s_X + s_Y
    if np.std(all_s) == 0:
        return 0.0
    
    effect_size = (np.mean(s_X) - np.mean(s_Y)) / np.std(all_s)
    return effect_size

# Define word sets
male_words = ['he', 'man', 'his', 'him', 'boy', 'male', 'father', 'son']
female_words = ['she', 'woman', 'her', 'girl', 'female', 'mother', 'daughter']
career_words = ['engineer', 'programmer', 'executive', 'manager', 'developer']
family_words = ['nurse', 'teacher', 'secretary', 'assistant', 'housekeeper']

wv_dict = {w: model.wv[w] for w in model.wv.key_to_index}
effect = compute_weat(male_words, female_words, career_words, family_words, wv_dict)

print(f"\n=== WEAT Effect Size ===")
print(f"Effect size: {effect:.4f}")
print("Interpretation:")
print(f"  {'Positive → male words more associated with career' if effect > 0 else 'Negative → female words more associated with career'}")
print(f"  |Effect| > 0.8 = strong bias")
```

## 4. Visualize Bias

```python
import matplotlib.pyplot as plt
from sklearn.decomposition import PCA

# Visualize gender direction in embedding space
words_to_plot = ['engineer', 'programmer', 'nurse', 'teacher',
                  'doctor', 'secretary', 'manager', 'assistant',
                  'he', 'she', 'man', 'woman']

valid_words = [w for w in words_to_plot if w in model.wv]
vecs = np.array([model.wv[w] for w in valid_words])

pca = PCA(n_components=1, random_state=42)
projections = pca.fit_transform(vecs)[:, 0]

# Sort by projection
sorted_pairs = sorted(zip(valid_words, projections), key=lambda x: x[1])

plt.figure(figsize=(10, 6))
colors = ['#3498db' if w in ['he', 'man'] else
          '#e74c3c' if w in ['she', 'woman', 'girl'] else
          '#f39c12' if projections[valid_words.index(w)] > 0 else '#2ecc71'
          for w in valid_words]

x_vals = [p for _, p in sorted_pairs]
y_vals = range(len(sorted_pairs))
labels = [w for w, _ in sorted_pairs]

plt.barh(y_vals, x_vals, color=colors[:len(sorted_pairs)])
plt.yticks(y_vals, labels)
plt.xlabel("Projection on gender direction")
plt.title("Gender Bias in Word Embeddings\n(right = masculine stereotypes)")
plt.axvline(x=0, color='black', linestyle='--', alpha=0.5)
plt.tight_layout()
plt.savefig("gender_bias.png", dpi=80)
plt.show()
print("Saved: gender_bias.png")
```

## 5. Types of Bias

| Bias Type | Example | Measurement |
|----------|---------|------------|
| Gender | doctor=male, nurse=female | WEAT, projection |
| Racial | Chỉ biểu hiện trong names | WEAT |
| Religious | Associations với religions | Similarity scores |
| Age | "old" vs "young" attributes | Analogy test |
| Geographic | Country stereotypes | Nearest neighbors |
