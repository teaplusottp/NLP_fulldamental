## 1. Word Alignment là gì?

**Word Alignment**: xác định các từ tương ứng nhau giữa hai ngôn ngữ trong câu song song.

```
EN: "The cat sat on the mat"
FR: "Le chat était assis sur le tapis"

Alignment:
  The  → Le
  cat  → chat  
  sat  → était assis (1-to-2 mapping)
  on   → sur
  the  → le
  mat  → tapis
```

## 2. Symmetrized Alignment

```python
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity

# IBM Model 2 simplified: alignment based on position + embedding sim
def compute_alignment(src_words: list, tgt_words: list,
                      src_vecs: dict, tgt_vecs: dict) -> list:
    """
    Compute word alignment between source and target sentences
    Returns: list of (src_idx, tgt_idx, score)
    """
    n = len(src_words)
    m = len(tgt_words)
    
    # Cosine similarity matrix
    sim_matrix = np.zeros((n, m))
    for i, sw in enumerate(src_words):
        for j, tw in enumerate(tgt_words):
            if sw in src_vecs and tw in tgt_vecs:
                sim = cosine_similarity([src_vecs[sw]], [tgt_vecs[tw]])[0][0]
                sim_matrix[i, j] = max(0, sim)
    
    # Find best alignment (greedy)
    alignments = []
    for i in range(n):
        best_j = np.argmax(sim_matrix[i])
        score = sim_matrix[i, best_j]
        if score > 0.3:
            alignments.append((i, best_j, score))
    
    return alignments, sim_matrix


# Demo với pretrained bilingual vectors
np.random.seed(42)
dim = 30

# Simulate cross-lingual embedding space
# Words that mean the same → similar vectors
bilingual_pairs = {
    ('cat', 'chat'): 0,
    ('dog', 'chien'): 1,
    ('sat', 'assis'): 2,
    ('the', 'le'): 3,
    ('on', 'sur'): 4,
    ('mat', 'tapis'): 5,
    ('was', 'était'): 6,
}

en_vecs = {}
fr_vecs = {}
cluster_centers = [np.random.randn(dim) for _ in range(7)]

for (en, fr), cluster_id in bilingual_pairs.items():
    center = cluster_centers[cluster_id]
    en_vecs[en] = center + np.random.randn(dim) * 0.1
    fr_vecs[fr] = center + np.random.randn(dim) * 0.1

en_sentence = ["the", "cat", "sat", "on", "the", "mat"]
fr_sentence = ["le", "chat", "était", "assis", "sur", "le", "tapis"]

alignments, sim_matrix = compute_alignment(en_sentence, fr_sentence, en_vecs, fr_vecs)

print("=== Word Alignment ===")
for src_idx, tgt_idx, score in alignments:
    src_word = en_sentence[src_idx]
    tgt_word = fr_sentence[tgt_idx]
    print(f"  '{src_word}' ↔ '{tgt_word}': {score:.4f}")

# Visualize
import matplotlib.pyplot as plt
fig, ax = plt.subplots(figsize=(10, 6))
im = ax.imshow(sim_matrix, cmap='Blues', aspect='auto')
ax.set_xticks(range(len(fr_sentence)))
ax.set_yticks(range(len(en_sentence)))
ax.set_xticklabels(fr_sentence, rotation=45)
ax.set_yticklabels(en_sentence)
ax.set_xlabel("French")
ax.set_ylabel("English")
ax.set_title("Word Alignment Matrix")
plt.colorbar(im)
plt.tight_layout()
plt.savefig("word_alignment.png", dpi=80)
plt.show()
```

## 3. Cross-lingual Sentence Retrieval

```python
# Task: Tìm câu EN tương ứng với câu FR (parallel corpus retrieval)

en_sentences = [
    "The dog is playing in the park",
    "The cat is sleeping on the sofa",
    "I enjoy reading books at home",
    "The weather is sunny and warm",
    "Machine learning transforms industries",
]

fr_sentences = [
    "Le chien joue dans le parc",
    "Le chat dort sur le canapé",
    "J'aime lire des livres à la maison",
    "Le temps est ensoleillé et chaud",
    "L'apprentissage automatique transforme les industries",
]

# Build sentence embeddings (mean of word vectors)
def mean_embed(tokens: list, vecs: dict) -> np.ndarray:
    relevant = [vecs[t] for t in tokens if t in vecs]
    return np.mean(relevant, axis=0) if relevant else np.zeros(dim)

# Tokenize (simplified)
def tokenize(text):
    import re
    return re.findall(r"[a-zàâçéèêëîïôûùüÿæœ']+", text.lower())

en_embeds = np.array([mean_embed(tokenize(s), en_vecs) for s in en_sentences])
fr_embeds = np.array([mean_embed(tokenize(s), fr_vecs) for s in fr_sentences])

# Cross-lingual retrieval: FR → EN
sim_matrix = cosine_similarity(fr_embeds, en_embeds)

print("\n=== Cross-lingual Sentence Retrieval ===")
correct = 0
for i, fr_sent in enumerate(fr_sentences):
    pred_en_idx = np.argmax(sim_matrix[i])
    match = "✅" if pred_en_idx == i else f"❌ (got: {pred_en_idx})"
    print(f"  FR: '{fr_sent[:40]}...'")
    print(f"  EN: '{en_sentences[pred_en_idx][:40]}...' {match}\n")
    if pred_en_idx == i:
        correct += 1

print(f"Retrieval accuracy: {correct}/{len(fr_sentences)}")
```

## 5. Tóm tắt

| Technique | Requires | Quality | Use Case |
|-----------|---------|---------|---------|
| Bilingual dictionary | Seed dict | Good | Translation |
| Procrustes | Parallel dict | Good | Alignment |
| MUSE | No supervision | Very Good | Unsupervised MT |
| mBERT | Large multilingual | Excellent | All cross-lingual NLU |
| LaBSE | Parallel sentences | Best (STS) | Cross-lingual search |

- Word alignment là core component của machine translation
- Cross-lingual embeddings enable zero-shot NLP across languages
- Modern: XLM-R/mBERT làm alignment tự động bên trong
