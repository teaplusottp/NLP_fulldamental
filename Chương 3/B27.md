## 1. Tại sao Debiasing quan trọng?

Bias trong embeddings → harm trong downstream applications:
- **HR systems**: CV screening thiên vị gender/race
- **Sentiment analysis**: từ liên quan đến group nhất định bị phân tích lệch
- **Translation**: "Doctor" → tiếng Việt luôn ra "bác sĩ nam"?
- **Search engines**: reinforcing stereotypes

## 2. Hard Debiasing (Bolukbasi et al. 2016)

```python
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity
from sklearn.decomposition import PCA
from gensim.models import Word2Vec
import warnings
warnings.filterwarnings('ignore')

# Train biased model
biased_corpus = [
    ["he", "engineer", "programmer", "doctor", "lawyer"],
    ["she", "nurse", "teacher", "secretary", "assistant"],
    ["man", "boss", "leader", "executive", "manager"],
    ["woman", "cook", "housekeeper", "receptionist"],
    ["male", "strong", "aggressive", "ambitious"],
    ["female", "gentle", "caring", "emotional"],
] * 30

model = Word2Vec(biased_corpus, vector_size=50, window=5,
                 min_count=1, sg=1, epochs=300, seed=42)

# ===== HARD DEBIASING =====

def identify_gender_direction(model) -> np.ndarray:
    """Find gender subspace via PCA on gender word pairs"""
    definitional_pairs = [
        ('he', 'she'), ('man', 'woman'), ('male', 'female'),
    ]
    
    pair_diffs = []
    for w1, w2 in definitional_pairs:
        if w1 in model.wv and w2 in model.wv:
            diff = model.wv[w1] - model.wv[w2]
            pair_diffs.append(diff)
    
    if not pair_diffs:
        return np.zeros(model.vector_size)
    
    matrix = np.array(pair_diffs)
    pca = PCA(n_components=1, random_state=42)
    pca.fit(matrix)
    gender_direction = pca.components_[0]
    return gender_direction / np.linalg.norm(gender_direction)

def neutralize(word: str, gender_direction: np.ndarray, model) -> np.ndarray:
    """Remove gender component from neutral words"""
    if word not in model.wv:
        return None
    vec = model.wv[word].copy()
    proj = np.dot(vec, gender_direction) * gender_direction
    return vec - proj  # Remove projection onto gender axis

def equalize(pair: tuple, gender_direction: np.ndarray, model) -> tuple:
    """Equalize gendered pairs around the midpoint"""
    w1, w2 = pair
    if w1 not in model.wv or w2 not in model.wv:
        return None, None
    
    v1, v2 = model.wv[w1].copy(), model.wv[w2].copy()
    midpoint = (v1 + v2) / 2
    
    # Neutralize midpoint
    mid_proj = np.dot(midpoint, gender_direction) * gender_direction
    mid_neutral = midpoint - mid_proj
    
    # Restore gender component
    v1_gender = np.dot(v1, gender_direction)
    v2_gender = np.dot(v2, gender_direction)
    
    v1_new = mid_neutral + v1_gender * gender_direction
    v2_new = mid_neutral + v2_gender * gender_direction
    
    return v1_new, v2_new

# Execute debiasing
gender_dir = identify_gender_direction(model)

# Target words to debias (should be gender-neutral)
neutral_words = ['engineer', 'programmer', 'nurse', 'teacher',
                  'doctor', 'lawyer', 'secretary', 'manager']

debiased_vecs = {}
for word in neutral_words:
    debiased = neutralize(word, gender_dir, model)
    if debiased is not None:
        debiased_vecs[word] = debiased

# Equalize pairs
equal_pairs = [('he', 'she'), ('man', 'woman')]

print("=== Gender Projection: Before → After Debiasing ===")
print(f"{'Word':12s} {'Before':>8} {'After':>8}")
print("-" * 32)
for word in neutral_words:
    if word in model.wv and word in debiased_vecs:
        before = np.dot(model.wv[word], gender_dir)
        after = np.dot(debiased_vecs[word], gender_dir)
        print(f"{word:12s} {before:8.4f} {after:8.4f}")
```

## 3. Soft Debiasing

```python
def soft_debias(word_vectors: dict, gender_dir: np.ndarray, 
                neutral_words: list, lambda_reg: float = 0.5) -> dict:
    """
    Soft debiasing: minimize gender projection while preserving similarity
    GloVe-based debiasing approach
    """
    debiased = {}
    
    for word, vec in word_vectors.items():
        if word in neutral_words:
            # Reduce (không loại bỏ hoàn toàn) gender component
            gender_proj = np.dot(vec, gender_dir) * gender_dir
            debiased[word] = vec - lambda_reg * gender_proj
        else:
            debiased[word] = vec.copy()
    
    return debiased

wv_dict = {w: model.wv[w].copy() for w in model.wv.key_to_index}
soft_debiased = soft_debias(wv_dict, gender_dir, neutral_words, lambda_reg=0.9)

print("\n=== Soft Debiasing Effect ===")
for word in ['engineer', 'nurse', 'doctor', 'teacher']:
    if word in soft_debiased:
        orig_proj = np.dot(model.wv[word], gender_dir)
        new_proj = np.dot(soft_debiased[word], gender_dir)
        pct_reduced = abs(1 - abs(new_proj) / (abs(orig_proj) + 1e-9)) * 100
        print(f"  {word:12s}: {orig_proj:+.4f} → {new_proj:+.4f} ({pct_reduced:.0f}% reduced)")
```

## 4. Evaluate Debiasing

```python
def measure_gender_bias(word: str, vecs: dict, gender_dir: np.ndarray) -> float:
    """Larger absolute value = more biased"""
    if word not in vecs:
        return 0.0
    return np.dot(vecs[word], gender_dir)

def utility_preservation(word: str, orig_vecs: dict, new_vecs: dict) -> float:
    """Cosine similarity between original and debiased (higher = more preserved)"""
    if word not in orig_vecs or word not in new_vecs:
        return 0.0
    return cosine_similarity([orig_vecs[word]], [new_vecs[word]])[0][0]

print("\n=== Evaluation: Bias vs Utility ===")
print(f"{'Word':12s} {'Orig Bias':>10} {'Debiased Bias':>14} {'Utility Preserved':>18}")
print("-" * 58)

orig_vecs = {w: model.wv[w] for w in model.wv.key_to_index}

for word in neutral_words:
    if word in debiased_vecs:
        orig_bias = measure_gender_bias(word, orig_vecs, gender_dir)
        new_bias = measure_gender_bias(word, debiased_vecs, gender_dir)
        utility = utility_preservation(word, orig_vecs, debiased_vecs)
        print(f"{word:12s} {orig_bias:10.4f} {new_bias:14.4f} {utility:18.4f}")
```

## 5. Limitations of Debiasing

```python
print("""
Limitations (Gonen & Goldberg 2019):
  1. Debiasing removes direct bias but leaves INDIRECT bias
     "doctor" debiased... but still closest to "surgeon", "specialist" (male-stereotyped)
     
  2. Bias clusters remain even after projection removal
     t-SNE still shows male professions clustering together
  
  3. Trade-off: more debiasing → more utility loss
  
Better approaches:
  - Include balanced data during pre-training
  - Counterfactual data augmentation (swap he↔she in training)
  - Fair representation learning (adversarial)
  - INLP (Iterative Nullspace Projection) - more complete debiasing
""")
```

## 6. Tóm tắt

| Method | Bias Removed | Utility Preserved | Complexity |
|--------|-------------|------------------|-----------|
| Hard Debiasing | Direct only | Good | Low |
| Soft Debiasing | Partial | Very Good | Low |
| INLP | More complete | Moderate | Medium |
| Re-training (fair) | Best | Best | High |

- Debiasing quan trọng cho fairness trong AI systems
- Hard debiasing: project ra khỏi gender subspace
- Không có silver bullet → combine multiple approaches
- Best: curate balanced training data từ đầu
