## 2. SIF (Smooth Inverse Frequency)

```python
import numpy as np
from collections import Counter
from typing import Optional

def compute_sif_embedding(sentences: list, word_vectors, a: float = 1e-3) -> np.ndarray:
    """
    SIF (Arora et al. 2017):
    1. Weighted sum: weight(w) = a / (a + p(w))
    2. Remove top-1 PC (principal component)
    
    Better than simple mean pooling for semantic similarity
    """
    # Tính word frequencies
    all_words = [w for s in sentences for w in s.lower().split()]
    word_count = Counter(all_words)
    total = sum(word_count.values())
    word_prob = {w: c/total for w, c in word_count.items()}
    
    dim = next(iter(word_vectors.values())).shape[0]
    sentence_embeddings = []
    
    for sentence in sentences:
        tokens = sentence.lower().split()
        vectors = []
        weights = []
        
        for word in tokens:
            if word in word_vectors:
                p = word_prob.get(word, 1e-6)
                weight = a / (a + p)
                vectors.append(weight * word_vectors[word])
                weights.append(weight)
        
        if not vectors:
            sentence_embeddings.append(np.zeros(dim))
        else:
            sentence_embeddings.append(np.mean(vectors, axis=0))
    
    embeddings = np.array(sentence_embeddings)
    
    # Remove principal component
    from sklearn.decomposition import TruncatedSVD
    if len(embeddings) > 1:
        svd = TruncatedSVD(n_components=1, random_state=42)
        svd.fit(embeddings)
        pc = svd.components_[0]
        # Remove the projection onto the first PC
        embeddings = embeddings - embeddings.dot(pc.T).reshape(-1, 1) * pc
    
    return embeddings


# Demo
from gensim.models import Word2Vec

corpus_train = [
    "the cat sat on the mat",
    "dogs are loyal and friendly",
    "python is a programming language",
    "machine learning is important",
    "cats and dogs are pets",
]

model = Word2Vec([s.split() for s in corpus_train],
                 vector_size=50, min_count=1, sg=1, epochs=300, seed=42)

# Build word_vectors dict
wv_dict = {w: model.wv[w] for w in model.wv.key_to_index}

# Compare sentences
sentences = [
    "cats and dogs are animals and pets",
    "python programming for data science",
    "cats are friendly pets",
    "machine learning algorithms",
]

sif_emb = compute_sif_embedding(sentences, wv_dict)
mean_emb = np.array([
    np.mean([wv_dict[w] for w in s.lower().split() if w in wv_dict], axis=0)
    if any(w in wv_dict for w in s.lower().split()) else np.zeros(50)
    for s in sentences
])

print("=== Sentence Similarity: SIF vs Mean Pooling ===")
from sklearn.metrics.pairwise import cosine_similarity

for i, s1 in enumerate(sentences):
    for j, s2 in enumerate(sentences):
        if j <= i:
            continue
        sif_sim = cosine_similarity([sif_emb[i]], [sif_emb[j]])[0][0]
        mean_sim = cosine_similarity([mean_emb[i]], [mean_emb[j]])[0][0]
        print(f"'{s1[:25]}' vs '{s2[:25]}'")
        print(f"  SIF: {sif_sim:.4f} | Mean: {mean_sim:.4f}")
```

## 3. Universal Sentence Encoder (USE)

```python
# USE - Google 2018
# Dùng Transformer encoder (large) hoặc Deep Averaging Network (lite)
# Output: 512-dim embedding

try:
    import tensorflow_hub as hub
    
    use_model = hub.load("https://tfhub.dev/google/universal-sentence-encoder/4")
    
    sentences = [
        "The weather is lovely today.",
        "It's so sunny outside!",
        "He drove to the airport.",
        "She is going to fly to London.",
    ]
    
    embeddings = use_model(sentences).numpy()
    print(f"USE embedding shape: {embeddings.shape}")  # (4, 512)
    
    sim_matrix = cosine_similarity(embeddings)
    print("\nSimilarity matrix:")
    for i, s1 in enumerate(sentences):
        for j, s2 in enumerate(sentences):
            if j > i:
                print(f"  '{s1[:20]}' vs '{s2[:20]}': {sim_matrix[i,j]:.4f}")

except ImportError:
    print("tensorflow_hub not installed")
    print("pip install tensorflow tensorflow_hub")
    print("""
Schematic USE similarities:
  'The weather is lovely' vs 'It's so sunny': 0.89 (paraphrase ✅)
  'The weather is lovely' vs 'He drove to airport': 0.23 (unrelated ✅)
""")
```

## 4. Sentence-BERT (SBERT)

```python
# SBERT - Reimers & Gurevych 2019
# BERT fine-tuned với Siamese network cho semantic similarity
# Output: dense sentence embeddings tốt hơn CLS token của BERT

try:
    from sentence_transformers import SentenceTransformer
    from sklearn.metrics.pairwise import cosine_similarity
    
    model_sbert = SentenceTransformer('all-MiniLM-L6-v2')

    sentences = [
        "A man is eating food.",
        "A man is eating a piece of bread.",
        "A horse is eating grass.",
        "The weather is nice outside.",
    ]
    
    embeddings = model_sbert.encode(sentences)
    print(f"SBERT shape: {embeddings.shape}")  # (4, 384)
    
    # Semantic search demo
    query = "someone is having a meal"
    query_emb = model_sbert.encode([query])
    
    sims = cosine_similarity(query_emb, embeddings)[0]
    print(f"\nSemantic search: '{query}'")
    for i, (sent, sim) in enumerate(sorted(zip(sentences, sims), key=lambda x: -x[1])):
        print(f"  {sim:.4f}: {sent}")

except ImportError:
    print("sentence_transformers not installed")
    print("pip install sentence-transformers")
    print("""
SBERT architecture:
  Input: sentence A → BERT → mean pool → embedding_A (384d)
  Input: sentence B → BERT → mean pool → embedding_B (384d)
  Training: |emb_A - emb_B| → cosine similarity → NLI labels
  → Much better semantic similarity than raw BERT CLS token
""")
```

## 6. Tóm tắt

- **SIF**: simple improvement over mean pooling (remove PC, IDF weight)
- **USE**: pre-trained, fast, good for production
- **SBERT**: best for semantic similarity/search tasks
- Chọn method dựa trên: quality need × speed requirement × infrastructure
