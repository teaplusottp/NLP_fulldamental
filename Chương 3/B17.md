## 1. Polysemy là gì?

**Polysemy**: một từ có nhiều nghĩa khác nhau.

```
"bank"    → ngân hàng / bờ sông / nghiêng máy bay
"fly"     → bay / con ruồi / khóa quần
"bright"  → sáng (ánh sáng) / thông minh / rực rỡ (màu sắc)
"cold"    → lạnh / cảm lạnh / lạnh lùng (tính cách)
"right"   → phải (hướng) / đúng / quyền lợi
```

corpus_finance = [
    ["bank", "money", "deposit", "savings", "loan", "credit"],
    ["bank", "finance", "invest", "profit", "account", "withdraw"],
    ["bank", "interest", "rate", "currency", "transaction"],
]
corpus_nature = [
    ["river", "bank", "water", "fish", "shore", "sand"],
    ["bank", "stream", "flood", "grass", "trees", "nature"],
    ["bank", "mud", "stone", "wildlife", "flowing", "current"],
]
corpus_combined = corpus_finance + corpus_nature

model_combined = Word2Vec(corpus_combined, vector_size=50, min_count=1,
                          sg=1, epochs=300, seed=42)

bank_vec = model_combined.wv["bank"]

# Tìm những từ gần nhất

for word, sim in model_combined.wv.most_similar("bank", topn=10):
    print(f"  {word:12s}: {sim:.4f}")

print("\n→ Trộn lẫn cả finance vocab lẫn nature vocab → vector không rõ ràng")

# So sánh similarity với 2 domain
finance_words = ["money", "deposit", "savings"]
nature_words = ["river", "water", "shore"]

finance_sims = [model_combined.wv.similarity("bank", w) for w in finance_words]

print(f"'bank' vs nature words:  {np.mean(nature_sims):.4f} avg")

## 3. Multi-Prototype Embedding (Huang et al. 2012)

```python
# Ý tưởng: mỗi từ có NHIỀU vectors (một per sense)
# Cluster context vectors → assign to senses

class MultiPrototypeEmbedding:
    """Simplified multi-prototype word embedding"""
    
    def __init__(self, n_prototypes=2):
        self.n_prototypes = n_prototypes
        self.prototypes = {}
    
    def fit(self, word: str, context_vectors: list):
        """Cluster context vectors into n_prototypes"""
        from sklearn.cluster import KMeans
        
        if len(context_vectors) < self.n_prototypes:
            self.prototypes[word] = [np.mean(context_vectors, axis=0)]
            return
        
        kmeans = KMeans(n_clusters=self.n_prototypes, random_state=42, n_init=10)
        labels = kmeans.fit_predict(context_vectors)
        
        self.prototypes[word] = [
            np.mean([v for v, l in zip(context_vectors, labels) if l == k], axis=0)
            for k in range(self.n_prototypes)
        ]
    
    def get_embedding(self, word: str, context_vector: np.ndarray) -> np.ndarray:
        """Chọn prototype gần nhất với context"""
        if word not in self.prototypes:
            return None
        
        sims = [
            cosine_similarity([context_vector], [proto])[0][0]
            for proto in self.prototypes[word]
        ]
        best_proto_idx = np.argmax(sims)
        return self.prototypes[word][best_proto_idx], best_proto_idx

# Simulate context vectors cho "bank"
np.random.seed(42)
# Finance contexts → vectors có component cao ở finance dimensions
finance_contexts = [np.random.randn(50) + np.array([2]*25 + [0]*25) for _ in range(20)]
# Nature contexts → vectors có component cao ở nature dimensions
nature_contexts  = [np.random.randn(50) + np.array([0]*25 + [2]*25) for _ in range(20)]
all_contexts = finance_contexts + nature_contexts

mp = MultiPrototypeEmbedding(n_prototypes=2)
mp.fit("bank", all_contexts)

# Test: dùng một context để chọn prototype
test_finance_ctx = np.random.randn(50) + np.array([2]*25 + [0]*25)
test_nature_ctx  = np.random.randn(50) + np.array([0]*25 + [2]*25)

_, proto1 = mp.get_embedding("bank", test_finance_ctx)
_, proto2 = mp.get_embedding("bank", test_nature_ctx)

print(f"\nMulti-Prototype 'bank':")
print(f"  Finance context → selects prototype {proto1}")
print(f"  Nature context  → selects prototype {proto2}")
print(f"  {'Different sense ✅' if proto1 != proto2 else 'Same sense (corpus too small)'}")
```

## 4. Sense Disambiguation Heuristics

```python
# Simple WSD (Word Sense Disambiguation) dùng context keywords
SENSE_KEYWORDS = {
    "bank_finance": ["money", "deposit", "loan", "credit", "savings", "withdraw", "account"],
    "bank_nature":  ["river", "water", "shore", "fish", "stream", "flood", "mud"],
    "bat_animal":   ["fly", "wing", "cave", "nocturnal", "mammal", "insect"],
    "bat_sports":   ["hit", "ball", "swing", "baseball", "cricket", "player"],
}

def disambiguate(word: str, context_tokens: list) -> str:
    """Identify which sense of 'word' is used based on context"""
    relevant_senses = {k: v for k, v in SENSE_KEYWORDS.items() if k.startswith(word)}
    
    scores = {}
    for sense, keywords in relevant_senses.items():
        overlap = len(set(context_tokens) & set(keywords))
        scores[sense] = overlap
    
    if max(scores.values()) == 0:
        return word + "_AMBIGUOUS"
    
    return max(scores, key=scores.get)

tests = [
    ("bank", ["I", "went", "to", "the", "bank", "to", "deposit", "money"]),
    ("bank", ["the", "river", "bank", "was", "flooded", "with", "water"]),
    ("bat",  ["the", "bat", "flew", "out", "of", "the", "cave", "at", "night"]),
    ("bat",  ["he", "swung", "the", "bat", "and", "hit", "the", "ball"]),
]

print("=== Simple WSD ===")
for word, context in tests:
    sense = disambiguate(word, context)
    print(f"  Context: '{' '.join(context[:6])}...'")
    print(f"  '{word}' → {sense}\n")
```

## 6. Tóm tắt

| Approach | Polysemy | Complexity | Year |
|---------|---------|-----------|------|
| Word2Vec | ❌ 1 vector | Low | 2013 |
| Multi-Prototype | ✅ K vectors | Medium | 2012 |
| ELMo | ✅ Context-aware | High | 2018 |
| BERT | ✅ Full context | Very high | 2018 |

- Polysemy là lý do chính để chuyển từ static → contextual embeddings
- Multi-prototype là cầu nối lịch sử
- BERT/modern LLMs giải quyết hoàn toàn vấn đề này
