## 2. Thu thập & Làm sạch Text

```python
import re
import string
import numpy as np
import unicodedata
from collections import Counter
import warnings
warnings.filterwarnings('ignore')

# Corpus thực (lấy từ NLTK nếu có, hoặc dùng demo)
def get_corpus():
    try:
        import nltk
        nltk.download('brown', quiet=True)

        sentences = brown.sents(categories=['news', 'editorial', 'reviews'])
        return [list(sent) for sent in sentences[:2000]]

        # Demo corpus
        return [

            "deep learning uses neural networks with multiple layers".split(),
            "natural language processing analyzes human text".split(),
            "word embeddings represent words as dense vectors".split(),
            "word2vec learns word representations from text context".split(),
            "the skip-gram model predicts context words from target".split(),
            "continuous bag of words uses context to predict target".split(),
            "negative sampling makes training word2vec efficient".split(),
            "pretrained embeddings transfer knowledge across tasks".split(),
            "sentence embeddings represent entire documents as vectors".split(),
        ] * 50

raw_sentences = get_corpus()
print(f"Raw corpus: {len(raw_sentences)} sentences")
```

## 3. Preprocessing Pipeline

```python
STOPWORDS = {
    'the', 'a', 'an', 'is', 'it', 'in', 'on', 'at', 'to', 'of',
    'and', 'or', 'but', 'for', 'not', 'with', 'by', 'from', 'as',
    'that', 'this', 'which', 'was', 'are', 'be', 'been', 'have', 'has',
    'do', 'did', 'will', 'would', 'could', 'should', 'may', 'might',
    'i', 'we', 'you', 'he', 'she', 'they', 'it',
}

def preprocess_sentence(tokens: list,
                         min_len: int = 2,
                         remove_stopwords: bool = True,
                         remove_punct: bool = True) -> list:
    """Normalize, lowercase, filter tokens"""
    processed = []
    for tok in tokens:
        # Lowercase
        tok = tok.lower()
        # Remove punctuation
        if remove_punct:
            tok = re.sub(r'[^a-z0-9\'-]', '', tok)
        # Skip empty, stopwords, too short
        if len(tok) < min_len:
            continue
        if remove_stopwords and tok in STOPWORDS:
            continue
        # Skip if all digits
        if tok.isdigit():
            continue
        processed.append(tok)
    return processed

processed_corpus = []
for sent in raw_sentences:
    clean = preprocess_sentence(sent)
    if len(clean) >= 3:  # Min 3 tokens
        processed_corpus.append(clean)

all_tokens = [t for sent in processed_corpus for t in sent]
vocab = Counter(all_tokens)
print(f"Processed corpus: {len(processed_corpus)} sentences")
print(f"Total tokens: {len(all_tokens)}")
print(f"Unique tokens: {len(vocab)}")
print(f"Top 10 words: {vocab.most_common(10)}")
```

## 4. Train Word2Vec

```python
from gensim.models import Word2Vec
import time

# Hyperparameter configurations
configs = {
    'CBOW_small': {
        'vector_size': 50, 'window': 5, 'min_count': 2,
        'sg': 0, 'hs': 0, 'negative': 5, 'epochs': 100
    },
    'SkipGram_medium': {
        'vector_size': 100, 'window': 7, 'min_count': 2,
        'sg': 1, 'hs': 0, 'negative': 10, 'epochs': 200
    },
}

trained_models = {}
for config_name, params in configs.items():
    print(f"\nTraining {config_name}...")
    start = time.time()
    
    model = Word2Vec(
        sentences=processed_corpus,
        workers=2,
        seed=42,
        **params
    )
    
    elapsed = time.time() - start
    print(f"  ✅ Trained in {elapsed:.1f}s | Vocab: {len(model.wv)}")
    trained_models[config_name] = model
```

## 5. Evaluate Embedding Quality

```python
def evaluate_model(model, name: str):
    print(f"\n=== {name} ===")
    
    vocab_words = list(model.wv.key_to_index.keys())
    
    # Test 1: Most similar words
    test_words = ['learning', 'neural', 'language', 'word']
    for word in test_words:
        if word in model.wv:
            similar = model.wv.most_similar(word, topn=3)
            print(f"  Similar to '{word}': {[w for w, _ in similar]}")
    
    # Test 2: Analogy (simplified since corpus is small)
    analogies = [
        ('machine', 'learning', 'deep'),
        ('word', 'embeddings', 'sentence'),
    ]
    print(f"\n  Analogies:")
    for a, b, c in analogies:
        if all(w in model.wv for w in [a, b, c]):
            results = model.wv.most_similar(
                positive=[b, c], negative=[a], topn=3
            )
            print(f"    {a}:{b} = {c}:? → {[r[0] for r in results]}")
        else:
            print(f"    Some words not in vocab: {a}, {b}, {c}")
    
    # Test 3: Pair similarity
    pairs = [
        ('learning', 'training'),
        ('network', 'model'),
        ('word', 'sentence'),
        ('language', 'text'),
    ]
    print(f"\n  Similarities:")
    for w1, w2 in pairs:
        if w1 in model.wv and w2 in model.wv:
            sim = model.wv.similarity(w1, w2)
            print(f"    '{w1}' vs '{w2}': {sim:.4f}")

for name, model in trained_models.items():
    evaluate_model(model, name)
```

## 6. Save & Load Model

```python
import os

# Save full model (bao gồm vocab, training state)
save_dir = "saved_models"
os.makedirs(save_dir, exist_ok=True)

best_model_name = 'SkipGram_medium'
best_model = trained_models[best_model_name]

# Option A: Full model (nếu cần tiếp tục training)
best_model.save(f"{save_dir}/word2vec_full.model")

# Option B: Word vectors only (nhỏ hơn, cho inference)
best_model.wv.save(f"{save_dir}/word_vectors.kv")

# Option C: Text format (compatible với GloVe tools)
best_model.wv.save_word2vec_format(f"{save_dir}/vectors.txt", binary=False)

print("Models saved!")

# Load
from gensim.models import KeyedVectors, Word2Vec

loaded_wv = KeyedVectors.load(f"{save_dir}/word_vectors.kv")
print(f"\nLoaded: {len(loaded_wv)} vectors")

if 'learning' in loaded_wv:
    print(f"  'learning' dim: {len(loaded_wv['learning'])}")
    print(f"  Similar: {[w for w, _ in loaded_wv.most_similar('learning', topn=3)]}")
```

## 7. Kết luận & Checklist

```
✅ Corpus preprocessing:
   - Lowercase, remove punctuation
   - Filter stopwords, short words
   - Minimum sentence length

✅ Model training:
   - Try both CBOW and Skip-gram
   - Tune: vector_size, window, negative, epochs
   - Use min_count ≥ 2 to reduce noise

✅ Evaluation:
   - Most similar words (qualitative)
   - Word pair similarity (quantitative)
   - Analogy test

✅ Save model:
   - .kv for inference
   - .model for continued training
   - .txt for compatibility
```
