## 1. Model trong Production thay đổi theo thời gian

Sau khi deploy, model **không ổn định mãi**:

```
Tháng 1: Accuracy = 93%
Tháng 3: Accuracy = 89%   ← Data drift!
Tháng 6: Accuracy = 82%   ← Khủng hoảng!
```

**Nguyên nhân:**
- **Data drift**: distribution của input thay đổi
- **Concept drift**: relationship giữa X và y thay đổi
- **Label drift**: định nghĩa label thay đổi

## 2. Monitoring Metrics

```python
import numpy as np
import pandas as pd
from datetime import datetime, timedelta
from collections import deque
import warnings
warnings.filterwarnings('ignore')

class ModelMonitor:
    """Simple model monitoring system"""
    
    def __init__(self, model, window_size=100):
        self.model = model
        self.window_size = window_size
        self.predictions = deque(maxlen=window_size)
        self.ground_truths = deque(maxlen=window_size)
        self.timestamps = deque(maxlen=window_size)
        self.input_features = deque(maxlen=window_size)
    
    def log_prediction(self, text, pred, true_label=None):
        """Log một prediction"""
        self.predictions.append(pred)
        self.timestamps.append(datetime.now())
        self.input_features.append({
            'char_count': len(text),
            'word_count': len(text.split())
        })
        if true_label is not None:
            self.ground_truths.append(true_label)
    
    def get_metrics(self):
        """Tính metrics trong window hiện tại"""
        metrics = {}
        
        if self.ground_truths:
            from sklearn.metrics import accuracy_score, f1_score
            preds = list(self.predictions)[:len(self.ground_truths)]
            truths = list(self.ground_truths)
            if len(preds) == len(truths):
                metrics['accuracy'] = accuracy_score(truths, preds)
                metrics['f1'] = f1_score(truths, preds, average='weighted')
        
        # Input distribution
        if self.input_features:
            features_df = pd.DataFrame(self.input_features)
            metrics['avg_char_count'] = features_df['char_count'].mean()
            metrics['avg_word_count'] = features_df['word_count'].mean()
        
        # Prediction distribution
        if self.predictions:
            preds_list = list(self.predictions)
            unique, counts = np.unique(preds_list, return_counts=True)
            metrics['prediction_distribution'] = dict(zip(unique.tolist(), counts.tolist()))
        
        metrics['sample_count'] = len(self.predictions)
        return metrics

# Sử dụng
from sklearn.pipeline import Pipeline
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.linear_model import LogisticRegression

# Train model
texts = ["spam free win", "hello friend", "prize winner", "meeting today", "buy now"]
labels = [1, 0, 1, 0, 1]
pipeline = Pipeline([('tfidf', TfidfVectorizer()), ('clf', LogisticRegression(max_iter=1000))])
pipeline.fit(texts, labels)

monitor = ModelMonitor(pipeline)

# Simulate predictions
test_texts = ["win money now", "project review", "limited offer", "team standup"]
test_labels = [1, 0, 1, 0]

for text, true_label in zip(test_texts, test_labels):
    pred = pipeline.predict([text])[0]
    monitor.log_prediction(text, pred, true_label)

print("Monitoring Metrics:", monitor.get_metrics())
```

## 3. Data Drift Detection

```python
from scipy.stats import ks_2samp
import numpy as np

def detect_data_drift(reference_data, current_data, threshold=0.05):
    """Kiểm tra data drift bằng KS test"""
    # reference_data: training distribution đặc trưng
    # current_data: production distribution hiện tại
    
    drift_features = {}
    
    for feature in ['char_count', 'word_count']:
        ref_vals = [len(t) if feature == 'char_count' else len(t.split())
                    for t in reference_data]
        curr_vals = [len(t) if feature == 'char_count' else len(t.split())
                     for t in current_data]
        
        statistic, p_value = ks_2samp(ref_vals, curr_vals)
        is_drift = p_value < threshold
        
        drift_features[feature] = {
            'ks_statistic': round(statistic, 4),
            'p_value': round(p_value, 4),
            'drift_detected': is_drift
        }
    
    return drift_features

# Test
train_texts = ["short text", "medium length text here", "this is a longer sentence for testing", "brief"]
prod_texts = ["this is a very very very long text that is quite different from training distribution",
              "another extremely lengthy production text that might cause drift detection issues"]

drift_report = detect_data_drift(train_texts, prod_texts)
print("\nDrift Detection:")
for feature, result in drift_report.items():
    status = "⚠️ DRIFT!" if result['drift_detected'] else "✅ OK"
    print(f"  {feature}: {status} (KS={result['ks_statistic']}, p={result['p_value']})")
```

## 4. A/B Testing

```python
import random
from collections import defaultdict

class ABTest:
    """Simple A/B testing framework"""
    
    def __init__(self, model_a, model_b, split_ratio=0.5):
        self.model_a = model_a
        self.model_b = model_b
        self.split = split_ratio
        self.results = defaultdict(list)
    
    def predict(self, text, user_id=None):
        """Route user to A or B model"""
        # Deterministic routing based on user_id
        if user_id:
            group = 'A' if hash(str(user_id)) % 2 == 0 else 'B'
        else:
            group = 'A' if random.random() < self.split else 'B'
        
        model = self.model_a if group == 'A' else self.model_b
        pred = model.predict([text])[0]
        
        return pred, group
    
    def log_result(self, group, prediction, is_correct):
        self.results[group].append(is_correct)
    
    def report(self):
        print("A/B Test Results:")
        for group, results in self.results.items():
            if results:
                accuracy = sum(results) / len(results)
                n = len(results)
                print(f"  Model {group}: Accuracy={accuracy:.4f} (n={n})")

# Tạo 2 model versions
pipeline_v1 = Pipeline([
    ('tfidf', TfidfVectorizer(ngram_range=(1,1))),
    ('clf', LogisticRegression(max_iter=1000))
])
pipeline_v2 = Pipeline([
    ('tfidf', TfidfVectorizer(ngram_range=(1,2))),  # bigram = cải tiến
    ('clf', LogisticRegression(max_iter=1000))
])

pipeline_v1.fit(texts, labels)
pipeline_v2.fit(texts, labels)

ab_test = ABTest(pipeline_v1, pipeline_v2)

# Simulate traffic
test_cases = [
    ("win free money now", 1),
    ("meeting at 3pm tomorrow", 0),
    ("prize winner selected", 1),
    ("team project review", 0)
]

for i, (text, true_label) in enumerate(test_cases):
    pred, group = ab_test.predict(text, user_id=f"user_{i}")
    is_correct = (pred == true_label)
    ab_test.log_result(group, pred, is_correct)

ab_test.report()
```

## 5. Alerts & Thresholds

```python
class AlertSystem:
    """Monitor và alert khi metrics vượt threshold"""
    
    def __init__(self):
        self.thresholds = {
            'accuracy_min': 0.85,
            'latency_max_ms': 200,
            'drift_threshold': 0.05
        }
        self.alerts = []
    
    def check_metrics(self, metrics):
        """Kiểm tra metrics và tạo alerts"""
        if 'accuracy' in metrics:
            if metrics['accuracy'] < self.thresholds['accuracy_min']:
                self.alerts.append({
                    'level': 'CRITICAL',
                    'metric': 'accuracy',
                    'value': metrics['accuracy'],
                    'threshold': self.thresholds['accuracy_min'],
                    'message': f"Accuracy {metrics['accuracy']:.4f} below threshold!"
                })
    
    def report_alerts(self):
        if not self.alerts:
            print("✅ No alerts - system healthy")
        else:
            for alert in self.alerts:
                icon = "🚨" if alert['level'] == 'CRITICAL' else "⚠️"
                print(f"{icon} [{alert['level']}] {alert['message']}")

# Test alert
alert_sys = AlertSystem()
alert_sys.check_metrics({'accuracy': 0.75})  # Below threshold
alert_sys.report_alerts()
```

## 6. Tóm tắt

- **Data drift**: input distribution thay đổi → cần detect và retrain
- **Monitor** accuracy, latency, prediction distribution
- **A/B Testing**: so sánh model versions trên real traffic
- **Alerts**: notification khi metrics drop
- **Retrain trigger**: khi drift detected hoặc performance drop
- Production NLP = Training + Deployment + Monitoring (continuous cycle)
