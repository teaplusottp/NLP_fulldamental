## 2. Lưu Model

```python
import joblib
from sklearn.pipeline import Pipeline
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.linear_model import LogisticRegression

# Train model
texts = ["win free money", "hello friend", "prize winner", "meeting today"]
labels = [1, 0, 1, 0]

pipeline = Pipeline([
    ('tfidf', TfidfVectorizer(ngram_range=(1,2))),
    ('clf', LogisticRegression(max_iter=1000, random_state=42))
])
pipeline.fit(texts, labels)

# Lưu tất cả trong một pipeline
joblib.dump(pipeline, 'spam_model.pkl')
print("Model saved!")

# Load lại
loaded_model = joblib.load('spam_model.pkl')
pred = loaded_model.predict(["win money free now!"])
print(f"Prediction: {'Spam' if pred[0] else 'Ham'}")
```

## 3. Tạo REST API với FastAPI

```python
# File: app.py
# pip install fastapi uvicorn

from fastapi import FastAPI
from pydantic import BaseModel
import joblib

app = FastAPI(title="Spam Classifier API", version="1.0")

# Load model khi khởi động server
model = joblib.load("spam_model.pkl")

class TextInput(BaseModel):
    text: str

class PredictionOutput(BaseModel):
    text: str
    prediction: str
    is_spam: bool
    confidence: float

@app.get("/")
def root():
    return {"message": "Spam Classifier API", "status": "running"}

@app.get("/health")
def health():
    return {"status": "healthy"}

@app.post("/predict", response_model=PredictionOutput)
def predict(input_data: TextInput):
    text = input_data.text
    
    pred = model.predict([text])[0]
    proba = model.predict_proba([text])[0]
    is_spam = bool(pred == 1)
    
    return PredictionOutput(
        text=text,
        prediction="SPAM" if is_spam else "HAM",
        is_spam=is_spam,
        confidence=float(max(proba))
    )

@app.post("/predict_batch")
def predict_batch(inputs: list[TextInput]):
    texts = [i.text for i in inputs]
    preds = model.predict(texts)
    probas = model.predict_proba(texts)
    
    results = []
    for text, pred, proba in zip(texts, preds, probas):
        results.append({
            "text": text,
            "is_spam": bool(pred == 1),
            "confidence": float(max(proba))
        })
    return results

# Chạy server:
# uvicorn app:app --host 0.0.0.0 --port 8000 --reload
```

## 5. Versioning và Model Registry

```python
import json
import datetime

def save_model_with_metadata(model, model_name, metrics, version=None):
    """Lưu model kèm metadata"""
    if version is None:
        version = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    
    model_path = f"{model_name}_v{version}.pkl"
    meta_path = f"{model_name}_v{version}_meta.json"
    
    # Lưu model
    joblib.dump(model, model_path)
    
    # Lưu metadata
    metadata = {
        "model_name": model_name,
        "version": version,
        "saved_at": str(datetime.datetime.now()),
        "metrics": metrics,
        "model_path": model_path
    }
    
    with open(meta_path, 'w') as f:
        json.dump(metadata, f, indent=2)
    
    print(f"Model saved: {model_path}")
    print(f"Metadata: {meta_path}")
    return model_path, meta_path

# Ví dụ sử dụng
metrics = {
    "accuracy": 0.92,
    "f1_weighted": 0.91,
    "test_size": 50
}
save_model_with_metadata(pipeline, "spam_classifier", metrics)
```

## 6. Input Validation & Error Handling

```python
# Bổ sung validation cho API
from fastapi import HTTPException

@app.post("/predict_safe")
def predict_safe(input_data: TextInput):
    text = input_data.text.strip()
    
    # Validate input
    if not text:
        raise HTTPException(status_code=400, detail="Text cannot be empty")
    
    if len(text) > 10000:
        raise HTTPException(status_code=400, detail="Text too long (max 10000 chars)")
    
    if len(text.split()) < 1:
        raise HTTPException(status_code=400, detail="Text must contain at least 1 word")
    
    try:
        pred = model.predict([text])[0]
        proba = model.predict_proba([text])[0]
        
        return {
            "prediction": "SPAM" if pred == 1 else "HAM",
            "confidence": float(max(proba))
        }
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Prediction failed: {str(e)}")
```

## 7. Logging

```python
import logging

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('api.log'),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger(__name__)

@app.post("/predict_logged")
def predict_logged(input_data: TextInput):
    logger.info(f"Prediction request: '{input_data.text[:50]}'")
    
    pred = model.predict([input_data.text])[0]
    result = "SPAM" if pred == 1 else "HAM"
    
    logger.info(f"Prediction result: {result}")
    return {"prediction": result}
```

## 8. Tóm tắt

```
Train → joblib.dump() → FastAPI → Validate → Predict → Log
```

- `joblib.dump/load`: lưu/load model (bao gồm cả pipeline)
- **FastAPI**: tạo REST API nhanh, đẹp, có docs tự động tại `/docs`
- Validate input trước khi predict
- Log requests để monitor
- Lưu metadata cùng model để tracking
