## 🧹 Preprocessing Nâng cao

```python
def preprocess_sentiment(text):
    """Clean text cho sentiment analysis"""
    # Lowercase
    text = text.lower()
    # Xử lý các từ viết tắt phổ biến
    contractions = {
        "won't": "will not", "can't": "cannot", "n't": " not",
        "i'm": "i am", "it's": "it is", "that's": "that is"
    }
    for contraction, expansion in contractions.items():
        text = text.replace(contraction, expansion)
    
    # Xử lý repetition "gooood" → "good"
    text = re.sub(r'(.)\1{2,}', r'\1\1', text)
    
    # Giữ lại dấu ! ? vì chúng quan trọng cho sentiment
    text = re.sub(r'[^a-z\s!?]', ' ', text)
    text = re.sub(r'\s+', ' ', text).strip()
    return text

df['text_clean'] = df['text'].apply(preprocess_sentiment)
print(df[['text', 'text_clean']].head(3).to_string())
```

## 📊 Feature Engineering cho Sentiment

```python
def extract_sentiment_features(text):
    """Features đặc thù cho sentiment"""
    return {
        'exclamation': text.count('!'),
        'question': text.count('?'),
        'uppercase_ratio': sum(c.isupper() for c in text) / (len(text) + 1),
        'ellipsis': text.count('...'),
        'word_count': len(text.split()),
        # Positive/Negative word count
        'pos_words': sum(1 for w in ['great','excellent','love','amazing','perfect','best']
                         if w in text.lower()),
        'neg_words': sum(1 for w in ['terrible','awful','worst','hate','boring','bad']
                         if w in text.lower()),
    }

sentiment_features = pd.DataFrame([extract_sentiment_features(t) for t in df['text']])
print("\nSentiment features sample:")
print(sentiment_features.head())
```

```python
# Train best model
best_pipeline = Pipeline([
    ('tfidf', TfidfVectorizer(ngram_range=(1,2))),
    ('clf', LogisticRegression(max_iter=1000, C=1.0))

# Xem top sentiment words
vec = best_pipeline.named_steps['tfidf']
model = best_pipeline.named_steps['clf']
feature_names = vec.get_feature_names_out()
coef = model.coef_[0]

top_pos_idx = np.argsort(coef)[-15:][::-1]
top_neg_idx = np.argsort(coef)[:15]

print("\n=== Top POSITIVE features ===")
for i in top_pos_idx:
    print(f"  (+{coef[i]:.3f}) '{feature_names[i]}'")

print("\n=== Top NEGATIVE features ===")
for i in top_neg_idx:
    print(f"  ({coef[i]:.3f}) '{feature_names[i]}'")

# Final evaluation
y_pred = best_pipeline.predict(X_test)
```

```python
def analyze_sentiment(text, model, preprocess_fn=None):
    if preprocess_fn:
        text = preprocess_fn(text)
    pred = model.predict([text])[0]
    proba = model.predict_proba([text])[0]
    
    sentiment_scale = proba[1]  # probability of positive
    
    if sentiment_scale > 0.7:
        label = "😊 Very Positive"
    elif sentiment_scale > 0.5:
        label = "🙂 Positive"
    elif sentiment_scale > 0.3:
        label = "🙁 Negative"
    else:
        label = "😠 Very Negative"
    
    return label, sentiment_scale

test_reviews = [

    "Completely awful, walked out halfway through."
]

print("=== Sentiment Analysis Demo ===")
for review in test_reviews:
```
