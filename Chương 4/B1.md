## 1. Tại sao Neural Network cho NLP?

Machine Learning truyền thống (Chương 2) cần **feature engineering thủ công**.  
Neural Networks tự học features từ raw data → end-to-end learning.

```
ML cũ: Text → [TF-IDF, bag-of-words, ...] → Logistic Regression
Neural: Text → [Embedding → Layers] → Output (features học tự động)
```

## 2. Perceptron: Đơn vị cơ bản

```python
import numpy as np
import matplotlib.pyplot as plt

class Perceptron:
    """Đơn giản nhất: một neuron, binary classification"""
    
    def __init__(self, learning_rate=0.01, n_epochs=100):
        self.lr = learning_rate
        self.n_epochs = n_epochs
        self.weights = None
        self.bias = None
    
    def fit(self, X: np.ndarray, y: np.ndarray):
        n_samples, n_features = X.shape
        self.weights = np.zeros(n_features)
        self.bias = 0
        
        for epoch in range(self.n_epochs):
            for xi, yi in zip(X, y):
                # Predict
                z = np.dot(xi, self.weights) + self.bias
                y_pred = 1 if z >= 0 else 0
                
                # Update (perceptron rule)
                delta = self.lr * (yi - y_pred)
                self.weights += delta * xi
                self.bias += delta
    
    def predict(self, X: np.ndarray) -> np.ndarray:
        z = X @ self.weights + self.bias
        return (z >= 0).astype(int)

# Demo: simple binary classification
np.random.seed(42)
X = np.random.randn(100, 2)
y = (X[:, 0] + X[:, 1] > 0).astype(int)

perc = Perceptron(learning_rate=0.01, n_epochs=50)
perc.fit(X, y)
preds = perc.predict(X)
acc = (preds == y).mean()
print(f"Perceptron accuracy: {acc:.2%}")
```

## 3. Fully Connected Layer (Dense)

```python
class DenseLayer:
    """Y = X @ W + b, với activation"""
    
    def __init__(self, input_dim: int, output_dim: int):
        # He initialization
        self.W = np.random.randn(input_dim, output_dim) * np.sqrt(2. / input_dim)
        self.b = np.zeros(output_dim)
    
    def forward(self, X: np.ndarray) -> np.ndarray:
        self.X = X  # cache for backprop
        return X @ self.W + self.b
    
    def backward(self, grad_out: np.ndarray, lr: float = 0.01) -> np.ndarray:
        # Gradients
        grad_W = self.X.T @ grad_out
        grad_b = grad_out.sum(axis=0)
        grad_X = grad_out @ self.W.T
        
        # Update
        self.W -= lr * grad_W
        self.b -= lr * grad_b
        
        return grad_X

# Test
layer = DenseLayer(input_dim=10, output_dim=5)
X_test = np.random.randn(4, 10)
Y_test = layer.forward(X_test)
print(f"Dense layer: {X_test.shape} → {Y_test.shape}")
```

## 4. Multi-layer Perceptron (MLP) cho Text Classification

```python
def relu(x):
    return np.maximum(0, x)

def relu_grad(x):
    return (x > 0).astype(float)

def sigmoid(x):
    return 1 / (1 + np.exp(-np.clip(x, -500, 500)))

def softmax(x):
    e_x = np.exp(x - x.max(axis=-1, keepdims=True))
    return e_x / e_x.sum(axis=-1, keepdims=True)

class SimpleMLP:
    """2-layer MLP cho text classification"""
    
    def __init__(self, input_dim, hidden_dim, output_dim, lr=0.01):
        self.lr = lr
        self.W1 = np.random.randn(input_dim, hidden_dim) * 0.1
        self.b1 = np.zeros(hidden_dim)
        self.W2 = np.random.randn(hidden_dim, output_dim) * 0.1
        self.b2 = np.zeros(output_dim)
    
    def forward(self, X):
        self.X = X
        self.z1 = X @ self.W1 + self.b1
        self.a1 = relu(self.z1)
        self.z2 = self.a1 @ self.W2 + self.b2
        self.probs = softmax(self.z2)
        return self.probs
    
    def backward(self, y_true):
        n = len(y_true)
        # One-hot
        one_hot = np.zeros_like(self.probs)
        one_hot[np.arange(n), y_true] = 1
        
        # Layer 2 gradient
        dz2 = (self.probs - one_hot) / n
        dW2 = self.a1.T @ dz2
        db2 = dz2.sum(axis=0)
        
        # Layer 1 gradient
        da1 = dz2 @ self.W2.T
        dz1 = da1 * relu_grad(self.z1)
        dW1 = self.X.T @ dz1
        db1 = dz1.sum(axis=0)
        
        # Update
        self.W2 -= self.lr * dW2
        self.b2 -= self.lr * db2
        self.W1 -= self.lr * dW1
        self.b1 -= self.lr * db1
    
    def cross_entropy_loss(self, y_true):
        probs = self.probs[np.arange(len(y_true)), y_true]
        return -np.mean(np.log(probs + 1e-9))

# Generate fake text features (TF-IDF style)
np.random.seed(42)
n_samples = 200
n_features = 50
n_classes = 3

X_data = np.random.randn(n_samples, n_features)
y_data = np.random.randint(0, n_classes, n_samples)

# Train
model_mlp = SimpleMLP(input_dim=n_features, hidden_dim=32, output_dim=n_classes, lr=0.05)
losses = []

for epoch in range(200):
    probs = model_mlp.forward(X_data)
    loss = model_mlp.cross_entropy_loss(y_data)
    losses.append(loss)
    model_mlp.backward(y_data)

# Evaluate
probs_final = model_mlp.forward(X_data)
preds_final = probs_final.argmax(axis=1)
accuracy = (preds_final == y_data).mean()
print(f"\nMLP training complete:")
print(f"  Final loss: {losses[-1]:.4f}")
print(f"  Training accuracy: {accuracy:.2%}")
```

## 5. Kiến trúc NLP Pipeline với Neural Network

```
Input Text
   ↓
Tokenization ("cat sat on mat" → [cat, sat, on, mat])
   ↓
Embedding Layer (word → dense vector)
   ↓
Hidden Layer(s) (feature extraction)
   ↓
Output Layer (sigmoid/softmax)
   ↓
Loss (BCE/CE) → Backpropagation
```

## 6. Tóm tắt

| Concept | Mô tả |
|---------|-------|
| Perceptron | Đơn vị neuron cơ bản |
| Dense Layer | Y = XW + b |
| Activation | ReLU, Sigmoid, Softmax |
| Forward pass | Tính output |
| Backward pass | Tính gradient, cập nhật trọng số |
| Loss | Cross-entropy cho classification |

- Neural networks = building blocks cho DL NLP
- Bài tiếp theo: Activation Functions chi tiết
